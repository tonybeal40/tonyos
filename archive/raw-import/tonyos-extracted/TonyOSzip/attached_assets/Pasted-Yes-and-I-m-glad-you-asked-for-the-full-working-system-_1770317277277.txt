from flask import Flask, send_file, render_template_string
from openpyxl import load_workbook
import os
import uuid
import random
from datetime import datetime, timedelta

app = Flask(__name__)

TEMPLATE = "templates/dashboard_template.xlsx"
EXPORT_DIR = "exports"

os.makedirs(EXPORT_DIR, exist_ok=True)

HTML = """
<!DOCTYPE html>
<html>
<head>
<title>Excel Dashboard Generator</title>
<style>
body {font-family: Arial; background:#111; color:white; padding:40px;}
.card {background:#222;padding:20px;border-radius:10px;}
button {padding:12px 24px;background:#00b894;color:white;border:none;}
</style>
</head>
<body>

<h2>Dashboard Generator</h2>

<div class="card">
<p>Click to generate a populated dashboard.</p>
<a href="/generate">
<button>Generate Dashboard</button>
</a>
</div>

</body>
</html>
"""

@app.route("/")
def home():
    return render_template_string(HTML)

# ---------- Sample Data Generator ----------
def generate_data(rows=20):

    data = []
    start = datetime.today()

    for i in range(rows):
        date = start - timedelta(days=i)
        product = f"SKU-{100+i}"
        inventory = random.randint(100,1000)
        velocity = random.randint(1,25)
        score = random.randint(60,100)

        data.append([
            date.strftime("%Y-%m-%d"),
            product,
            inventory,
            velocity,
            score
        ])

    return data

# ---------- Dashboard Generator ----------
@app.route("/generate")
def generate_dashboard():

    wb = load_workbook(TEMPLATE)
    ws = wb["Data"]

    # Clear old rows (keep header)
    ws.delete_rows(2, ws.max_row)

    # Inject dataset
    dataset = generate_data()

    for row in dataset:
        ws.append(row)

    # Save export
    filename = f"{uuid.uuid4()}.xlsx"
    path = os.path.join(EXPORT_DIR, filename)

    wb.save(path)

    return send_file(path, as_attachment=True)

app.run(host="0.0.0.0", port=3000)

Any template will populate automatically — only if the template is wired correctly.

The code doesn’t magically understand Excel dashboards. It just pushes rows into a table. The dashboard updates because Excel formulas, pivots, and charts are pointing at that table.
