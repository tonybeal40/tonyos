<script>
function savePDF() {
  alert(
    "In the next window:\n\n" +
    "1. Destination → Save as PDF\n" +
    "2. Click Save\n\n" +
    "The file will be saved to your computer."
  );
  window.print();
}

document.getElementById("csvFile").addEventListener("change", function(e) {
  const reader = new FileReader();
  reader.onload = () => build(reader.result);
  reader.readAsText(e.target.files[0]);
});

function build(text) {
  const rows = text.trim().split("\n").map(r => r.split(","));
  const headers = rows[0].map(h => h.trim().toLowerCase());
  const data = rows.slice(1);

  const out = document.getElementById("output");
  out.innerHTML = "";

  const v = (row, key) => row[headers.indexOf(key)] || "";

  data.forEach(row => {
    const page = document.createElement("div");
    page.className = "company-page";

    page.innerHTML = `
      <div class="header">
        <div class="company">${v(row,"company")}</div>
        <div class="location">${v(row,"city")}, ${v(row,"state")}</div>
      </div>

      <table>
        <tr><th>Website</th><td>${v(row,"website")}</td></tr>
        <tr><th>Phone</th><td>${v(row,"phone")}</td></tr>
        <tr><th>Confidence</th><td>${v(row,"confidence")}</td></tr>
        <tr><th>Natoli Angle</th><td>${v(row,"natoli_angle")}</td></tr>
        <tr><th>Press OEM</th><td>${v(row,"press_oem")}</td></tr>
        <tr><th>Probable Defects</th><td>${v(row,"probable_defects")}</td></tr>
        <tr><th>Production Stage</th><td>${v(row,"production_stage")}</td></tr>
        <tr><th>Tooling Class</th><td>${v(row,"tooling_class")}</td></tr>
      </table>
    `;

    out.appendChild(page);
  });
}
</script>
