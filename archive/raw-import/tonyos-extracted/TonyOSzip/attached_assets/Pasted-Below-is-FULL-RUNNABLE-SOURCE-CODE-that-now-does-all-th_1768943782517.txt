from docx import Document
from docx.shared import Pt
from datetime import date

# =============================
# CONFIGURATION (EDIT THESE)
# =============================
COMPANY_NAME = "Your Company Name"
SYSTEM_NAME = "Tool Management II (TM-II)"
DOC_NUMBER = "SDS-TMII-001"
VERSION = "2025.6"
AUTHOR = "Tony Beal"
TODAY = date.today().isoformat()

OUTPUT_FILE = f"TMII_SDS_{COMPANY_NAME.replace(' ', '_')}_{DOC_NUMBER}_v{VERSION}.docx"

# =============================
# DOCUMENT SETUP
# =============================
doc = Document()

def heading(text, level=1):
    h = doc.add_heading(text, level=level)
    h.style.font.size = Pt(12)

def paragraph(text):
    p = doc.add_paragraph(text)
    p.style.font.size = Pt(10)

# =============================
# TITLE PAGE
# =============================
doc.add_heading("Software Design Specification", 0)
doc.add_heading(SYSTEM_NAME, 1)

paragraph(f"Company Name: {COMPANY_NAME}")
paragraph(f"Document Number: {DOC_NUMBER}")
paragraph(f"Document Type: Software Design Specification (SDS)")
paragraph(f"Version: {VERSION}")
paragraph("Status: Final – Validation Package")
paragraph("")
paragraph(f"Prepared By: {AUTHOR}")
paragraph(f"Date: {TODAY}")
paragraph("")
paragraph("Approved By:")
paragraph("System Owner: __________________________")
paragraph("Quality Assurance: ______________________")
paragraph("Validation: ____________________________")
paragraph("Management: ____________________________")

doc.add_page_break()

# =============================
# REVISION HISTORY
# =============================
heading("Revision History")

rev_table = doc.add_table(rows=1, cols=5)
hdr = rev_table.rows[0].cells
headers = ["Revision", "Date", "Description of Change", "Author", "Approval"]
for i, h in enumerate(headers):
    hdr[i].text = h

rev_rows = [
    ("1.0", TODAY, "Initial SDS created from approved IQ/OQ and User Manual", AUTHOR, "Pending"),
    ("1.1", TODAY, "Added traceability matrix and validation-safe enhancements", AUTHOR, "Pending")
]

for row in rev_rows:
    cells = rev_table.add_row().cells
    for i, value in enumerate(row):
        cells[i].text = value

doc.add_page_break()

# =============================
# SDS CONTENT
# =============================
heading("1. Purpose")
paragraph(
    "This Software Design Specification (SDS) defines the functional, architectural, "
    "and operational design of the Tool Management II (TM-II) software system. "
    "It establishes the design baseline supporting Installation Qualification (IQ), "
    "Operational Qualification (OQ), and validated use under cGMP."
)

heading("2. Scope")
paragraph(
    "This SDS applies to the TM-II application, TM-II Settings application, Magic Software XPA "
    "runtime, Microsoft SQL Server database, and associated licensed hardware and interfaces."
)

paragraph(
    "Structural code testing is excluded due to the use of Magic Software XPA as a validated "
    "commercial off-the-shelf (COTS) platform."
)

heading("3. System Overview")
paragraph(
    "TM-II provides controlled management of tooling inventory, inspections, usage tracking, "
    "audit trails, and reporting for tablet manufacturing environments."
)

heading("4. Functional Design")
heading("4.1 Tool Directory", 2)
paragraph(
    "The Tool Directory serves as the master entity for tooling records, including identification, "
    "quantity tracking, inspection tolerances, status, and drawing associations."
)

heading("4.2 Sets", 2)
paragraph(
    "Sets group tools used together in manufacturing while maintaining independent tool records."
)

heading("4.3 Products", 2)
paragraph(
    "Products define manufactured tablets and apply default usage and inspection limits."
)

heading("4.4 Conceptual Data Model", 2)
paragraph(
    "TM-II uses a relational data structure consisting of master (Tool Directory), relational "
    "(Sets, Products), and transactional entities (Inspections, Usage History, Audit Records)."
)

heading("5. Security Design")
paragraph(
    "User authentication, role-based authorization, and HASP hardware licensing control system access."
)

heading("6. Audit Trail and Data Integrity")
paragraph(
    "The system records user ID, date, and time for applicable actions and prevents unauthorized modifications."
)

heading("7. Error Handling and Recovery")
paragraph(
    "Authorization errors, license enforcement errors, and data validation errors are handled "
    "with controlled messaging. Backup and recovery processes are validated."
)

heading("8. Configuration and Code Boundary")
paragraph(
    "System behavior is governed by configuration tables. Core logic resides within the Magic XPA "
    "runtime and is not modified during normal operation."
)

heading("9. Design Assumptions")
paragraph(
    "System time is controlled by the OS. Database backups follow IT SOPs. Drawing files are managed "
    "outside application logic."
)

heading("10. Reference Documentation")
paragraph(
    "User operation is described in the Tool Management II User Reference Manual (Version 2025) "
    "and is consistent with this SDS."
)

doc.add_page_break()

# =============================
# TRACEABILITY MATRIX
# =============================
heading("11. Design → IQ → OQ Traceability Matrix")

table = doc.add_table(rows=1, cols=6)
hdr = table.rows[0].cells
headers = [
    "SDS Section",
    "Design Feature",
    "IQ Reference",
    "OQ Reference",
    "Verification Method",
    "Acceptance Criteria"
]

for i, h in enumerate(headers):
    hdr[i].text = h

matrix_rows = [
    ("4.1", "Tool Directory Management", "N/A", "8.2.4", "Functional Test", "Tools tracked correctly"),
    ("4.2", "Set Assembly", "N/A", "8.2.5", "Functional Test", "Sets managed correctly"),
    ("5", "Authentication & Authorization", "N/A", "8.1", "Security Test", "Unauthorized access blocked"),
    ("5", "HASP License Enforcement", "7.2", "8.1.6", "Hardware Test", "Execution blocked without key"),
    ("6", "Audit Trail", "N/A", "8.1.5.24", "Functional Test", "Audit records retained"),
    ("7", "Backup and Recovery", "7.1.3.2", "8.7", "Execution Test", "Recovery successful")
]

for row in matrix_rows:
    cells = table.add_row().cells
    for i, value in enumerate(row):
        cells[i].text = value

doc.add_page_break()

# =============================
# SDS vNEXT – CHANGE IMPACT (SAFE)
# =============================
heading("12. Change Impact Assessment – SDS vNext")

paragraph(
    "This section documents potential future revisions to this SDS. "
    "Changes described herein are documentation-only and do not alter system functionality."
)

paragraph(
    "Examples of acceptable SDS vNext changes include clarification of design intent, "
    "expanded descriptions, formatting updates, and traceability improvements."
)

paragraph(
    "Any changes affecting system behavior, configuration, or interfaces shall be "
    "evaluated through formal change control and may require revalidation."
)

# =============================
# SAVE FILE
# =============================
doc.save(OUTPUT_FILE)
print(f"SDS package generated successfully: {OUTPUT_FILE}")
