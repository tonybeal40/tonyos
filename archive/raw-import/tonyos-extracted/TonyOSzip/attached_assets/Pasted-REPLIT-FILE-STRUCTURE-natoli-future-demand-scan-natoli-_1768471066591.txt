const axios = require("axios");
const cheerio = require("cheerio");
const fs = require("fs");

const truths = JSON.parse(fs.readFileSync("natoli_truths.json", "utf8"));
const signals = JSON.parse(fs.readFileSync("future_demand_signals.json", "utf8"));

const visited = new Set();
const results = [];

function normalize(text) {
  return text.replace(/\s+/g, " ").toLowerCase();
}

function hasAny(text, list = []) {
  return list.some(k => text.includes(k));
}

async function fetchPage(url) {
  try {
    const res = await axios.get(url, { timeout: 10000 });
    const $ = cheerio.load(res.data);
    return normalize($("body").text());
  } catch {
    return "";
  }
}

// Focused crawl: homepage → about → careers → news
async function crawlCompany(baseUrl) {
  const paths = ["", "/about", "/careers", "/jobs", "/news"];
  let combinedText = "";

  for (const path of paths) {
    const url = baseUrl + path;
    if (visited.has(url)) continue;
    visited.add(url);

    const text = await fetchPage(url);
    if (text) combinedText += " " + text;
  }

  return combinedText;
}

function evaluateFutureNeed(text) {
  const hit = {
    hiring: hasAny(text, signals.future_need_signals.hiring),
    infrastructure: hasAny(text, signals.future_need_signals.infrastructure),
    language_shift: hasAny(text, signals.future_need_signals.language_shifts),
    excluded: hasAny(text, signals.future_need_signals.exclusions)
  };

  return hit;
}

function evaluateNatoliTruths(text) {
  let passCount = 0;

  for (const truth of truths.truths) {
    if (truth.required_keywords && hasAny(text, truth.required_keywords)) {
      passCount++;
    }
  }

  return passCount;
}

(async () => {
  // Seed list can come from earlier discovery outputs
  const seedCompanies = JSON.parse(
    fs.readFileSync("emerging_natoli_companies.json", "utf8")
  );

  for (const company of seedCompanies) {
    const text = await crawlCompany(company.url);
    if (!text) continue;

    if (hasAny(text, truths.global_blocked_keywords)) continue;

    const futureSignals = evaluateFutureNeed(text);
    if (futureSignals.excluded) continue;

    const truthScore = evaluateNatoliTruths(text);

    if (
      (futureSignals.hiring || futureSignals.infrastructure || futureSignals.language_shift) &&
      truthScore >= Math.floor(truths.truths.length / 2)
    ) {
      results.push({
        url: company.url,
        future_need_detected: true,
        indicators: futureSignals,
        natoli_truth_score: truthScore,
        verdict:
          "Company shows leading indicators of future tooling and formulation needs"
      });
    }
  }

  fs.writeFileSync(
    "future_natoli_targets.json",
    JSON.stringify(results, null, 2)
  );

  console.log(`Future-demand scan complete. ${results.length} targets identified.`);
})();
