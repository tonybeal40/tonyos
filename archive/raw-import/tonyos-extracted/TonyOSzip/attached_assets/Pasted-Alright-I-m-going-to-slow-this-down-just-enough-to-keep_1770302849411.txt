🧠 LAYER 2 — TonyOS Brain
brain/tonyLLM.js
import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

export async function tonyBrain(input) {

  const system = `
You are TonyOS Manufacturing Intelligence Engine.

Evaluate:
- compression relevance
- scale pressure
- tooling gaps
- validation exposure
- competitor gravity

Return JSON:
{
  compression: "",
  pressure: [],
  risk: [],
  competitors: [],
  decision: ""
}
`;

  const res = await client.chat.completions.create({
    model: "gpt-4.1-mini",
    messages: [
      { role: "system", content: system },
      { role: "user", content: input }
    ],
    temperature: 0
  });

  return JSON.parse(res.choices[0].message.content);
}

discovery/batchScan.js
import { tonyBrain } from "../brain/tonyLLM.js";

export async function batchScan(list) {

  const results = [];

  for (let item of list) {

    const analysis = await tonyBrain(item.description);

    results.push({
      name: item.name,
      analysis
    });
  }

  return results;
}

  );

async function scan() {

  const data = JSON.parse(document.getElementById("input").value);

  const res = await fetch("/scan", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify(data)
  });

  const result = await res.json();

  document.getElementById("output").innerText =
    JSON.stringify(result, null, 2);
}
