script.js
let pipelineTotal = 0;
let accounts = [];
let repTotals = {
"Northeast": 0,
"Southeast": 0,
"Midwest": 0,
"West": 0
};

function addAccount() {

let company = document.getElementById("company").value;
let stage = document.getElementById("stage").value;
let rep = document.getElementById("rep").value;
let value = parseInt(document.getElementById("value").value);
let next = document.getElementById("nextAction").value;

if (!company || !value) return;

accounts.push({company, stage, rep, value, next});
pipelineTotal += value;
repTotals[rep] += value;

updateTable();
updateDashboard();

document.querySelectorAll("input").forEach(i => i.value = "");
}

function updateTable() {

let table = document.getElementById("crmTable");
table.innerHTML = `
<tr>
<th>Account</th>
<th>Stage</th>
<th>Rep</th>
<th>Value</th>
<th>Next Action</th>
</tr>
`;

accounts.forEach(a => {

table.innerHTML += `
<tr>
<td>${a.company}</td>
<td>${a.stage}</td>
<td>${a.rep}</td>
<td>$${a.value}</td>
<td>${a.next}</td>
</tr>`;
});
}

function updateDashboard() {

document.getElementById("pipeline").innerText = pipelineTotal;
document.getElementById("count").innerText = accounts.length;

drawChart();
}

function drawChart() {

let canvas = document.getElementById("repChart");
let ctx = canvas.getContext("2d");

canvas.width = 350;
canvas.height = 200;

ctx.clearRect(0,0,canvas.width,canvas.height);

let reps = Object.keys(repTotals);
let max = Math.max(...Object.values(repTotals),1);

reps.forEach((rep,i)=>{

let height = (repTotals[rep]/max)*150;
ctx.fillStyle = "red";
ctx.fillRect(i*80+40,180-height,40,height);

ctx.fillStyle = "white";
ctx.fillText(rep,i*80+40,195);
});
}

function moneyFromValue(val){
  // convert dropdown value to numeric for totals
  const map = {
    "$10k": 10000,
    "$25k": 25000,
    "$50k": 50000,
    "$100k": 100000,
    "$250k": 250000,
    "$500k": 500000,
    "$1M": 1000000,
    "$2M+": 2000000
  };
  return map[val] ?? 0;
}

function probToNum(p){
  return parseInt(String(p).replace("%",""),10) / 100;
}

function weekKey(){
  // simple "YYYY-W" based on local date
  const d = new Date();
  const firstJan = new Date(d.getFullYear(),0,1);
  const days = Math.floor((d - firstJan) / 86400000);
  const week = Math.ceil((days + firstJan.getDay() + 1) / 7);
  return `${d.getFullYear()}-W${week}`;
}

function upsertAccountFromForm(payload){

  let a = STORE.accounts.find(x=>x.account === payload.account);
  if(!a){

function computeDashboard(){
  // totals
  const pipeline = STORE.opportunities.reduce((sum,o)=>sum + moneyFromValue(o.value), 0);
  const forecast = STORE.opportunities.reduce((sum,o)=>sum + moneyFromValue(o.value)*probToNum(o.prob), 0);

  // stale
  const staleCount = STORE.accounts.filter(a=>a.stale).length;

  // tier 1 coverage: % of tier 1 with active next action + due
  const tier1 = STORE.accounts.filter(a=>a.tier === "Tier 1");
  const tier1Covered = tier1.filter(a=>a.nextAction && a.due).length;
  const coverage = tier1.length ? Math.round((tier1Covered / tier1.length) * 100) : 0;

  // weekly touches
  const wk = weekKey();
  const touches = STORE.activity.filter(a=>a.weekKey === wk).length;

  $("kpiPipeline").textContent = `$${Math.round(pipeline).toLocaleString()}`;
  $("kpiForecast").textContent = `$${Math.round(forecast).toLocaleString()}`;
  $("kpiStale").textContent = String(staleCount);
  $("kpiCoverage").textContent = `${coverage}%`;
  $("kpiTouches").textContent = String(touches);

  // stage totals
  const stageTotals = {};
  OPTIONS.stages.forEach(s=>stageTotals[s]=0);
  STORE.opportunities.forEach(o=>{
    stageTotals[o.stage] = (stageTotals[o.stage] ?? 0) + moneyFromValue(o.value);
  });

  // rep totals
  const repTotals = {};
  OPTIONS.reps.forEach(r=>repTotals[r]=0);
  STORE.opportunities.forEach(o=>{
    repTotals[o.owner] = (repTotals[o.owner] ?? 0) + moneyFromValue(o.value);
  });

  // activity by rep (week)
  const actTotals = {};
  OPTIONS.reps.forEach(r=>actTotals[r]=0);
  STORE.activity.filter(a=>a.weekKey === wk).forEach(a=>{
    actTotals[a.rep] = (actTotals[a.rep] ?? 0) + 1;
  });

  renderBars("stageBars", stageTotals, pipeline || 1, true);
  renderBars("repBars", repTotals, pipeline || 1, true);
  renderBars("activityBars", actTotals, Math.max(...Object.values(actTotals), 1), false);

  renderRisks(stageTotals, repTotals, staleCount, coverage);
}

function renderBars(containerId, totals, maxBase, moneyMode){
  const el = $(containerId);
  el.innerHTML = "";

  const entries = Object.entries(totals)
    .sort((a,b)=>b[1]-a[1]);

  entries.forEach(([name,val])=>{
    const pct = Math.round((val / maxBase) * 100);
    const bar = document.createElement("div");
    bar.className = "bar";
    bar.innerHTML = `
      <div class="barTop">
        <div class="barName">${name}</div>
        <div class="barVal">${moneyMode ? "$"+Math.round(val).toLocaleString() : val}</div>
      </div>
      <div class="meter"><div class="fill" style="width:${Math.min(pct,100)}%"></div></div>
    `;
    el.appendChild(bar);
  });
}

function renderRisks(stageTotals, repTotals, staleCount, coverage){
  const list = $("riskList");
  list.innerHTML = "";

  // simple “CEO-friendly” risk statements
  const quoteVal = stageTotals["Quote"] ?? 0;
  const technicalVal = stageTotals["Technical"] ?? 0;
  const engagedVal = stageTotals["Engaged"] ?? 0;

  const topRep = Object.entries(repTotals).sort((a,b)=>b[1]-a[1])[0];
  const topRepName = topRep ? topRep[0] : "N/A";
  const topRepVal = topRep ? topRep[1] : 0;

  const risks = [];

  if(staleCount > 0){
    risks.push({t:`${staleCount} stale accounts`, d:"Accounts with 30+ days since last touch are at risk of silent loss"});
  }
  if(coverage < 70){
    risks.push({t:`Tier 1 coverage at ${coverage}%`, d:"Tier 1 accounts without next steps are invisible risk"});
  }
  if(quoteVal > 0 && technicalVal === 0 && engagedVal === 0){
    risks.push({t:"Pipeline is late-stage heavy", d:"Good for forecast, risky if top deals slip"});
  }
  if(topRepVal > 0){
    risks.push({t:`Concentration: ${topRepName} holds $${Math.round(topRepVal).toLocaleString()}`, d:"If one territory slips, forecast shifts materially"});
  }
  if(risks.length === 0){
    risks.push({t:"No major red flags detected", d:"Keep weekly updates consistent so the dashboard stays truthful"});
  }

  risks.forEach(r=>{
    const li = document.createElement("li");
    li.innerHTML = `<b>${r.t}</b> — ${r.d}`;
    list.appendChild(li);
  });
}

let deals = [];

function show(id){
document.querySelectorAll(".view").forEach(v=>v.classList.add("hidden"));
document.getElementById(id).classList.remove("hidden");
}

function save(){

let deal = {
id: Date.now(),
account: account.value,
stage: stage.value,
rep: rep.value,
action: action.value,
value: parseInt(value.value),
timestamp: Date.now()
};

deals.push(deal);

status.innerText = "Deal saved.";

render();
}

function render(){

renderTable();
renderKanban();
renderDashboard();
}

function renderTable(){

let table = document.getElementById("table");

table.innerHTML = `
<tr>
<th>Account</th>
<th>Stage</th>
<th>Rep</th>
<th>Value</th>
<th>Action</th>
</tr>`;

deals.forEach(d=>{

let stale = Date.now()-d.timestamp > 60000;

table.innerHTML += `
<tr style="${stale?'background:#552222':''}">
<td>${d.account}</td>
<td>${d.stage}</td>
<td>${d.rep}</td>
<td>$${d.value}</td>
<td>${d.action}</td>
</tr>`;
});
}

function renderKanban(){

["Identified","Engaged","Technical","Quote"].forEach(stage=>{
document.getElementById(stage).innerHTML="";
});

deals.forEach(d=>{

let stale = Date.now()-d.timestamp > 60000;

let card = document.createElement("div");
card.className = "card "+(stale?"stale":"");
card.draggable = true;
card.id = d.id;

card.innerText = `${d.account} ($${d.value})`;

card.ondragstart = drag;

document.getElementById(d.stage).appendChild(card);
});
}

function renderDashboard(){

let total = 0;
let reps = {};

deals.forEach(d=>{
total += d.value;
reps[d.rep] = (reps[d.rep]||0)+d.value;
});

pipeline.innerText = total;
count.innerText = deals.length;

let repStats = document.getElementById("repStats");
repStats.innerHTML="";

for(let r in reps){
repStats.innerHTML += `<div>${r}: $${reps[r]}</div>`;
}
}

function allowDrop(ev){ev.preventDefault();}

function drag(ev){ev.dataTransfer.setData("text",ev.target.id);}

function drop(ev){

ev.preventDefault();

let id = ev.dataTransfer.getData("text");
let card = document.getElementById(id);

let newStage = ev.currentTarget.querySelector("div").id;

let deal = deals.find(d=>d.id==id);

deal.stage = ev.currentTarget.querySelector("h3").innerText;

render();
}

function exportCSV(){

let csv = "Account,Stage,Rep,Value,Action\n";

deals.forEach(d=>{
csv += `${d.account},${d.stage},${d.rep},${d.value},${d.action}\n`;
});

let blob = new Blob([csv]);
let a = document.createElement("a");
a.href = URL.createObjectURL(blob);
a.download = "natoli_crm_export.csv";
a.click();
}
