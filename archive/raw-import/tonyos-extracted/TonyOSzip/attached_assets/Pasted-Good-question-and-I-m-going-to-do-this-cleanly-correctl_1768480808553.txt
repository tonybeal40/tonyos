2️⃣ competitor_press_scan.js (FULL SOURCE CODE)

👉 Paste exactly

const axios = require("axios");
const cheerio = require("cheerio");
const fs = require("fs");

const competitors = JSON.parse(
  fs.readFileSync("competitor_presses.json", "utf8")
);

const SEARCH_PHRASES = competitors.presses.flatMap(p =>
  p.models.map(m => `${p.company} ${m}`)
);

const results = [];
const visited = new Set();

function normalize(text) {
  return text.replace(/\s+/g, " ").toLowerCase();
}

// DuckDuckGo HTML search
async function searchWeb(query) {
  const url = `https://duckduckgo.com/html/?q=${encodeURIComponent(query)}`;
  const res = await axios.get(url, { timeout: 10000 });
  const $ = cheerio.load(res.data);

  const links = new Set();
  $("a.result__a").each((_, el) => {
    const href = $(el).attr("href");
    if (href && href.startsWith("http")) {
      links.add(href.split("&")[0]);
    }
  });

  return Array.from(links);
}

// Crawl limited company pages
async function crawlCompany(baseUrl) {
  const paths = ["", "/careers", "/jobs", "/about", "/news"];
  let text = "";

  for (const path of paths) {
    const url = baseUrl + path;
    if (visited.has(url)) continue;
    visited.add(url);

    try {
      const res = await axios.get(url, { timeout: 10000 });
      const $ = cheerio.load(res.data);
      text += " " + $("body").text();
    } catch {}
  }

  return normalize(text);
}

// Detect competitor press mentions
function detectPressSignals(text) {
  const matches = [];

  for (const press of competitors.presses) {
    for (const model of press.models) {
      const phrase = `${press.company} ${model}`.toLowerCase();
      if (text.includes(phrase)) {
        matches.push({
          manufacturer: press.company,
          model: model
        });
      }
    }
  }

  return matches;
}

// Confidence scoring
function confidenceLevel(matchCount) {
  if (matchCount >= 3) return "High confidence signal";
  if (matchCount === 2) return "Medium confidence signal";
  return "Low confidence signal";
}

(async () => {
  for (const phrase of SEARCH_PHRASES) {
    console.log(`Searching: ${phrase}`);
    const links = await searchWeb(phrase);

    for (const link of links) {
      const text = await crawlCompany(link);
      if (!text) continue;

      const matches = detectPressSignals(text);
      if (matches.length === 0) continue;

      results.push({
        company_url: link,
        detected_presses: matches,
        confidence: confidenceLevel(matches.length),
        detected_via: phrase
      });
    }
  }

  fs.writeFileSync(
    "competitor_press_signals.json",
    JSON.stringify(results, null, 2)
  );

  console.log(`Scan complete. ${results.length} companies flagged.`);
})();
