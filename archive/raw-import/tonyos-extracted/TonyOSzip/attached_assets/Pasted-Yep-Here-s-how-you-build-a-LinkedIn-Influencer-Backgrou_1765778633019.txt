This is basically Canva, but stripped down to the 20 percent that actually matters for LinkedIn

What the app does (MVP that wins)

You open the app and you can generate:

Scroll stopping LinkedIn post backgrounds (1080x1080, 1080x1350, 1200x627)

Clean quote cards, list cards, “framework” cards, and “metric proof” cards

A LinkedIn style post preview (so you see how it will look in-feed)

One click export to PNG

No fluff. No 1,000 buttons. Just elite templates and typography

The influencer look (what you’re really asking for)

Influencer visuals on LinkedIn usually follow these rules:

Big type, strong hierarchy (headline huge, details small)

1 idea per card

Simple layout grids

High contrast backgrounds (dark gradients or clean light panels)

One accent color only

Lots of spacing

We’ll bake those rules into the templates so you can’t mess it up

app.js
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

const formatEl = document.getElementById("format");
const templateEl = document.getElementById("template");
const themeEl = document.getElementById("theme");
const headlineEl = document.getElementById("headline");
const bodyEl = document.getElementById("body");
const accentEl = document.getElementById("accent");
const noiseEl = document.getElementById("noise");
const sigEl = document.getElementById("signature");

const btnExport = document.getElementById("btnExport");
const btnRandom = document.getElementById("btnRandom");

const previewName = document.getElementById("previewName");
const previewPng = document.getElementById("previewPng");

function setCanvasSize(){
  const f = formatEl.value;
  if (f === "square"){ canvas.width = 1080; canvas.height = 1080; }
  if (f === "portrait"){ canvas.width = 1080; canvas.height = 1350; }
  if (f === "landscape"){ canvas.width = 1200; canvas.height = 627; }
}

function themes(accent){
  return {
    midnight: {
      bg: ["#07070a", "#111436"],
      panel: "rgba(255,255,255,.06)",
      ink: "#f5f7ff",
      muted: "rgba(245,247,255,.70)",
      accent
    },
    neonSlate: {
      bg: ["#07070a", "#0b1b2c"],
      panel: "rgba(255,255,255,.06)",
      ink: "#f5f7ff",
      muted: "rgba(245,247,255,.70)",
      accent
    },
    cleanLight: {
      bg: ["#f7f7fb", "#ffffff"],
      panel: "rgba(0,0,0,.06)",
      ink: "#0b0f1a",
      muted: "rgba(11,15,26,.65)",
      accent
    },
    steel: {
      bg: ["#0b0f1a", "#151a2a"],
      panel: "rgba(255,255,255,.06)",
      ink: "#f5f7ff",
      muted: "rgba(245,247,255,.70)",
      accent
    }
  };
}

function drawGradient(t){
  const g = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);
  g.addColorStop(0, t.bg[0]);
  g.addColorStop(1, t.bg[1]);
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  const glow = ctx.createRadialGradient(canvas.width*0.75, canvas.height*0.2, 0, canvas.width*0.75, canvas.height*0.2, canvas.width*0.6);
  glow.addColorStop(0, hexToRgba(t.accent, 0.22));
  glow.addColorStop(1, "rgba(0,0,0,0)");
  ctx.fillStyle = glow;
  ctx.fillRect(0, 0, canvas.width, canvas.height);
}

function drawNoise(amount){
  const a = Number(amount);
  if (!a) return;

  const img = ctx.getImageData(0, 0, canvas.width, canvas.height);
  const d = img.data;
  for (let i = 0; i < d.length; i += 4) {
    const n = (Math.random() - 0.5) * a;
    d[i] = clamp(d[i] + n, 0, 255);
    d[i+1] = clamp(d[i+1] + n, 0, 255);
    d[i+2] = clamp(d[i+2] + n, 0, 255);
  }
  ctx.putImageData(img, 0, 0);
}

function drawRoundedPanel(x, y, w, h, r, fill){
  ctx.fillStyle = fill;
  ctx.beginPath();
  roundRect(ctx, x, y, w, h, r);
  ctx.fill();
}

function render(){
  setCanvasSize();

  const accent = accentEl.value;
  const t = themes(accent)[themeEl.value];

  drawGradient(t);

  const pad = Math.round(canvas.width * 0.075);
  const panelW = canvas.width - pad*2;
  const panelH = canvas.height - pad*2;

  drawRoundedPanel(pad, pad, panelW, panelH, 42, t.panel);

  ctx.fillStyle = t.ink;
  ctx.textAlign = "left";

  const headline = headlineEl.value.trim();
  const body = bodyEl.value.trim();
  const sig = sigEl.value.trim();

  if (templateEl.value === "statement") drawStatement(t, pad, headline, body, sig);
  if (templateEl.value === "list") drawList(t, pad, headline, body, sig);
  if (templateEl.value === "framework") drawFramework(t, pad, headline, body, sig);
  if (templateEl.value === "metric") drawMetric(t, pad, headline, body, sig);

  drawNoise(noiseEl.value);

  previewName.textContent = sig || "Your Name";
  previewPng.src = canvas.toDataURL("image/png");
}

function drawStatement(t, pad, headline, body, sig){
  const x = pad + 56;
  const y = pad + 86;

  ctx.fillStyle = t.accent;
  ctx.fillRect(x, y - 16, 90, 8);

  ctx.fillStyle = t.ink;
  ctx.font = "800 76px ui-sans-serif, system-ui";
  wrapText(headline || "Write your headline", x, y + 70, canvas.width - (pad*2) - 112, 86);

  ctx.fillStyle = t.muted;
  ctx.font = "500 38px ui-sans-serif, system-ui";
  wrapText(body || "", x, y + 300, canvas.width - (pad*2) - 112, 52);

  signatureLine(t, sig, pad);
}

function drawList(t, pad, headline, body, sig){
  const x = pad + 56;
  const y = pad + 86;

  ctx.fillStyle = t.ink;
  ctx.font = "800 64px ui-sans-serif, system-ui";
  wrapText(headline || "3 things that matter", x, y + 60, canvas.width - (pad*2) - 112, 76);

  const items = (body || "One\nTwo\nThree").split("\n").filter(Boolean).slice(0, 4);
  let yy = y + 220;

  ctx.font = "600 40px ui-sans-serif, system-ui";
  items.forEach((it, idx) => {
    const bulletX = x;
    ctx.fillStyle = t.accent;
    ctx.beginPath();
    ctx.arc(bulletX + 10, yy - 10, 10, 0, Math.PI*2);
    ctx.fill();

    ctx.fillStyle = t.ink;
    ctx.fillText(it.trim(), bulletX + 36, yy);
    yy += 78;
  });

  signatureLine(t, sig, pad);
}

function drawFramework(t, pad, headline, body, sig){
  const x = pad + 56;
  const y = pad + 86;

  ctx.fillStyle = t.ink;
  ctx.font = "800 58px ui-sans-serif, system-ui";
  wrapText(headline || "Framework", x, y + 55, canvas.width - (pad*2) - 112, 68);

  const cells = (body || "Point 1\nPoint 2\nPoint 3\nPoint 4").split("\n").filter(Boolean).slice(0, 4);
  const gridTop = y + 150;

  const gap = 24;
  const w = (canvas.width - (pad*2) - 112 - gap) / 2;
  const h = (canvas.height - (pad*2) - 260 - gap) / 2;

  let i = 0;
  for (let r = 0; r < 2; r++){
    for (let c = 0; c < 2; c++){
      const bx = x + c*(w + gap);
      const by = gridTop + r*(h + gap);

      drawRoundedPanel(bx, by, w, h, 26, "rgba(0,0,0,.18)");

      ctx.fillStyle = t.accent;
      ctx.fillRect(bx + 26, by + 28, 70, 6);

      ctx.fillStyle = t.ink;
      ctx.font = "700 36px ui-sans-serif, system-ui";
      wrapText(cells[i] || "", bx + 26, by + 80, w - 52, 46);
      i++;
    }
  }

  signatureLine(t, sig, pad);
}

function drawMetric(t, pad, headline, body, sig){
  const x = pad + 56;
  const y = pad + 86;

  const metric = (headline || "77").trim();
  const label = (body || "Your SSI score, if you care about real proof").trim();

  ctx.fillStyle = t.accent;
  ctx.font = "900 160px ui-sans-serif, system-ui";
  ctx.fillText(metric, x, y + 180);

  ctx.fillStyle = t.ink;
  ctx.font = "800 54px ui-sans-serif, system-ui";
  wrapText("Proof beats opinions", x, y + 260, canvas.width - (pad*2) - 112, 64);

  ctx.fillStyle = t.muted;
  ctx.font = "600 38px ui-sans-serif, system-ui";
  wrapText(label, x, y + 340, canvas.width - (pad*2) - 112, 52);

  signatureLine(t, sig, pad);
}

function signatureLine(t, sig, pad){
  if (!sig) return;
  ctx.fillStyle = t.muted;
  ctx.font = "650 34px ui-sans-serif, system-ui";
  ctx.textAlign = "left";
  ctx.fillText(sig, pad + 56, canvas.height - pad - 56);
}

function wrapText(text, x, y, maxWidth, lineHeight){
  const words = text.split(" ");
  let line = "";
  let yy = y;

  for (let n = 0; n < words.length; n++){
    const testLine = line + words[n] + " ";
    const w = ctx.measureText(testLine).width;
    if (w > maxWidth && n > 0){
      ctx.fillText(line, x, yy);
      line = words[n] + " ";
      yy += lineHeight;
    } else {
      line = testLine;
    }
  }
  ctx.fillText(line, x, yy);
}

function hexToRgba(hex, a){
  const v = hex.replace("#","");
  const r = parseInt(v.slice(0,2),16);
  const g = parseInt(v.slice(2,4),16);
  const b = parseInt(v.slice(4,6),16);
  return `rgba(${r},${g},${b},${a})`;
}

function clamp(v,a,b){ return Math.max(a, Math.min(b, v)); }

function roundRect(ctx, x, y, w, h, r){
  const rr = Math.min(r, w/2, h/2);
  ctx.moveTo(x+rr, y);
  ctx.arcTo(x+w, y, x+w, y+h, rr);
  ctx.arcTo(x+w, y+h, x, y+h, rr);
  ctx.arcTo(x, y+h, x, y, rr);
  ctx.arcTo(x, y, x+w, y, rr);
  ctx.closePath();
}

function randomize(){
  const themes = ["midnight", "neonSlate", "cleanLight", "steel"];
  const templates = ["statement", "list", "framework", "metric"];
  themeEl.value = themes[Math.floor(Math.random() * themes.length)];
  templateEl.value = templates[Math.floor(Math.random() * templates.length)];
  accentEl.value = ["#ff2d55","#22c55e","#7c3aed","#06b6d4"][Math.floor(Math.random() * 4)];
  noiseEl.value = String(Math.floor(Math.random() * 18));
  render();
}

btnExport.addEventListener("click", () => {
  const a = document.createElement("a");
  a.download = "linkedin-card.png";
  a.href = canvas.toDataURL("image/png");
  a.click();
});

btnRandom.addEventListener("click", randomize);

[
  formatEl, templateEl, themeEl,
  headlineEl, bodyEl, accentEl,
  noiseEl, sigEl
].forEach(el => el.addEventListener("input", render));

setCanvasSize();
render();
