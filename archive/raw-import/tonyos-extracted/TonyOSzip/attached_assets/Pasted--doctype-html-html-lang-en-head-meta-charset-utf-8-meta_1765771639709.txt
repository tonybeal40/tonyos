  <script>
    // -------------------------
    // State + helpers
    // -------------------------
    const $ = (id) => document.getElementById(id);

    const state = {
      truthLock: true,
      history: []
    };

    function nowTime(){
      const d = new Date();
      return d.toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'});
    }

    function setOnline(ok){
      const dot = $("onlineDot");
      const txt = $("onlineTxt");
      dot.classList.remove("on","off");
      if(ok){
        dot.classList.add("on");
        txt.textContent = "online";
      } else {
        dot.classList.add("off");
        txt.textContent = "offline";
      }
    }

    function updateBadges(){
      $("badgeMode").textContent = $("mode").value;
      $("badgeMood").textContent = $("mood").value;
      $("badgeSession").textContent = $("sessionId").value || "default";
    }

    function addMsg(role, text, meta={}){
      const wrap = document.createElement("div");
      wrap.className = "msg " + (role === "user" ? "user" : "ai");

      const top = document.createElement("div");
      top.className = "metaRow";

      const left = document.createElement("div");
      left.textContent = `${role === "user" ? "You" : "TonyOS"} · ${nowTime()}`;

      const right = document.createElement("div");
      right.className = "smallBtns";

      const copyBtn = document.createElement("button");
      copyBtn.className = "btn tiny";
      copyBtn.textContent = "Copy";
      copyBtn.onclick = async () => {
        await navigator.clipboard.writeText(text);
        copyBtn.textContent = "Copied";
        setTimeout(()=>copyBtn.textContent="Copy", 900);
      };

      right.appendChild(copyBtn);
      top.appendChild(left);
      top.appendChild(right);

      const content = document.createElement("div");
      content.className = "content";
      content.textContent = text;

      wrap.appendChild(top);
      wrap.appendChild(content);

      if(meta && Object.keys(meta).length){
        const metaDiv = document.createElement("div");
        metaDiv.className = "help mono";
        metaDiv.style.marginTop = "8px";
        metaDiv.textContent = JSON.stringify(meta, null, 2);
        wrap.appendChild(metaDiv);
      }

      $("log").appendChild(wrap);
      $("log").scrollTop = $("log").scrollHeight;

      state.history.push({role, text, meta, ts: Date.now()});
    }

    async function ping(){
      const base = ($("serverUrl").value || "").trim().replace(/\/+$/, "");
      if(!base) return setOnline(false);

      try{
        const r = await fetch(base + "/health", { method:"GET" });
        setOnline(r.ok);
      } catch(e){
        setOnline(false);
      }
    }

    function parseUrls(){
      const raw = $("urls").value.trim();
      if(!raw) return null;
      return raw.split(",").map(s => s.trim()).filter(Boolean);
    }

    async function sendMessage(withResearch){
      updateBadges();

      const base = ($("serverUrl").value || "").trim().replace(/\/+$/, "");
      const session_id = ($("sessionId").value || "default").trim();
      let mode = $("mode").value;
      const mood = $("mood").value;

      // Truth lock forces mode = truth for claims
      if(state.truthLock && mode !== "truth"){
        // Don’t silently override when user chose code lane. Only override if not code.
        if(mode !== "code") mode = "truth";
      }

      const message = $("msg").value.trim();
      if(!message) return;

      addMsg("user", message, {mode, mood});

      $("msg").value = "";
      await ping();

      const payload = {
        message,
        mode,
        mood,
        session_id,
        research_urls: withResearch ? (parseUrls() || []) : null
      };

      try{
        const r = await fetch(base + "/chat", {
          method:"POST",
          headers: {"Content-Type":"application/json"},
          body: JSON.stringify(payload)
        });

        if(!r.ok){
          const t = await r.text();
          addMsg("ai", "Server error: " + t);
          return;
        }

        const data = await r.json();
        const answer = data.answer || "(no answer)";
        const citations = data.citations || [];
        addMsg("ai", answer, {citations_count: citations.length});

      } catch(e){
        addMsg("ai", "Offline. Start your local server first. Error: " + e.message);
        setOnline(false);
      }
    }

    // -------------------------
    // Events
    // -------------------------
    $("btnPing").onclick = ping;
    $("btnSend").onclick = () => sendMessage(false);
    $("btnSendResearch").onclick = () => sendMessage(true);

    $("btnClear").onclick = () => {
      $("log").innerHTML = "";
      state.history = [];
    };

    $("btnExport").onclick = () => {
      const blob = new Blob([JSON.stringify(state.history, null, 2)], {type:"application/json"});
      const a = document.createElement("a");
      a.href = URL.createObjectURL(blob);
      a.download = "tonyos_ai_chat_export.json";
      a.click();
    };

    $("truthLock").onclick = () => {
      state.truthLock = !state.truthLock;
      $("truthLock").classList.toggle("on", state.truthLock);
    };

    ["mode","mood","sessionId"].forEach(id => $(id).addEventListener("change", updateBadges));
    updateBadges();
    ping();
    setInterval(ping, 8000);

    // Enter to send (Ctrl+Enter)
    $("msg").addEventListener("keydown", (e) => {
      if(e.key === "Enter" && (e.ctrlKey || e.metaKey)){
        e.preventDefault();
        sendMessage(false);
      }
    });
  </script>
