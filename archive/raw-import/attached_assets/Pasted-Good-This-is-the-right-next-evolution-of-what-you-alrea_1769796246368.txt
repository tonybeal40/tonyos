app.js (CORE LOGIC)
function runSearch() {
  const company = document.getElementById("company").value;
  const product = document.getElementById("productType").value;
  const results = document.getElementById("results");

  let score = 0;
  let evidence = [];

  signals.forEach(signal => {
    if (Math.random() > 0.5) { // placeholder logic
      score += signal.weight;
      evidence.push(signal.name);
    }
  });

  let confidence =
    score >= 8 ? "High" :
    score >= 4 ? "Medium" :
    "Low";

  results.innerHTML = `
    <strong>Company:</strong> ${company}<br>
    <strong>Product:</strong> ${product}<br><br>
    <strong>Likely Manufacturing Model:</strong> ${
      evidence.includes("Manufactured For Language") ? "Outsourced CDMO" : "In-house or Hybrid"
    }<br>
    <strong>Evidence Signals:</strong><br>
    ${evidence.map(e => `• ${e}`).join("<br>")}<br><br>
    <strong>Confidence Level:</strong> ${confidence}<br>
    <strong>Natoli Relevance:</strong> ${
      product === "Tablet" && confidence !== "Low"
        ? "Strong potential tooling / press relevance"
        : "Limited or indirect fit"
    }
  `;

  results.classList.remove("hidden");
}
