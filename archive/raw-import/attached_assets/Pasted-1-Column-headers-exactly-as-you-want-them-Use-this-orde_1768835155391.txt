3️⃣ Python code to generate and populate this structure

Below is a clean Python script that:

Creates your table

✅ Python: Natoli Account Structuring Script
import pandas as pd

# -----------------------------
# INPUT: raw company data
# -----------------------------
# Expected columns if available:
# Company, City, State, Phone, Website, LinkedIn, industry, website_text

input_file = "raw_companies.csv"
output_file = "natoli_targets.csv"

df = pd.read_csv(input_file)

# -----------------------------
# Natoli Fit Logic
# -----------------------------

def determine_natoli_fit(text):
    if not isinstance(text, str):
        return "No Fit", "Insufficient public manufacturing signal"

    t = text.lower()

    engineering_signals = [
        "tablet", "capsule", "solid oral", "manufacturing",
        "production", "batch", "throughput", "press"
    ]

    scientific_signals = [
        "formulation", "ms&t", "process development",
        "analytical", "characterization", "scale-up"
    ]

    disqualifiers = [
        "injectable only", "infusion", "admixture",
        "clinic", "wellness", "pellet", "trt"
    ]

    if any(d in t for d in disqualifiers):
        return "No Fit", "Primarily sterile, clinical, or non-compression operation"

    eng = any(e in t for e in engineering_signals)
    sci = any(s in t for s in scientific_signals)

    if eng and sci:
        return "Both", "Production-scale compression with active development or scale-up"
    elif eng:
        return "Engineering", "Tablet production where press uptime and tooling reliability matter"
    elif sci:
        return "Scientific", "Focus on formulation, compaction understanding, or scale-up"
    else:
        return "No Fit", "No clear tablet compression or compaction signal"

# -----------------------------
# Apply logic
# -----------------------------

fits = df["website_text"].apply(determine_natoli_fit)
df["Natoli Fit"] = fits.apply(lambda x: x[0])
df["Reason"] = fits.apply(lambda x: x[1])

# -----------------------------
# Ensure final columns
# -----------------------------

final_columns = [
    "Company",
    "City",
    "State",
    "Phone",
    "Contact",
    "Website",
    "LinkedIn",
    "Natoli Fit",
    "Reason"
]

for col in final_columns:
    if col not in df.columns:
        df[col] = ""

df = df[final_columns]

# -----------------------------
# Output
# -----------------------------

df.to_csv(output_file, index=False)
print(f"Natoli target file created: {output_file}")
