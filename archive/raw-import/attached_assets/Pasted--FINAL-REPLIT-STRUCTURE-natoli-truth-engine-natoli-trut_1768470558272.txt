2️⃣ truth_engine.js

Binary truth enforcement + rejection language + routing

const fs = require("fs");

const rules = JSON.parse(fs.readFileSync("natoli_truths.json", "utf8"));
const companies = JSON.parse(fs.readFileSync("input_companies.json", "utf8"));

function normalize(text) {
  return text.toLowerCase();
}

function hasAny(text, list = []) {
  return list.some(k => text.includes(k));
}

function evaluateCompany(company) {
  const text = normalize(
    `${company.name} ${company.industry} ${company.description}`
  );

  const result = {
    name: company.name,
    qualified: true,
    routing: [],
    failed_truths: [],
    rejection_reasons: []
  };

  // GLOBAL HARD BLOCK
  if (hasAny(text, rules.global_blocked_keywords)) {
    result.qualified = false;
    result.rejection_reasons.push(
      "Company operates in a globally excluded manufacturing category for Natoli."
    );
    return result;
  }

  // TRUTH CHECKS
  for (const truth of rules.truths) {
    if (truth.required_keywords && !hasAny(text, truth.required_keywords)) {
      result.qualified = false;
      result.failed_truths.push(truth.id);
      result.rejection_reasons.push(truth.rejection_sentence);
    }

    if (truth.blocked_keywords && hasAny(text, truth.blocked_keywords)) {
      result.qualified = false;
      result.failed_truths.push(truth.id);
      result.rejection_reasons.push(truth.rejection_sentence);
    }
  }

  // ROUTING (ONLY IF QUALIFIED)
  if (result.qualified) {
    for (const [route, keywords] of Object.entries(rules.routing_rules)) {
      if (hasAny(text, keywords)) {
        result.routing.push(route);
      }
    }

    if (result.routing.length === 0) {
      result.routing.push("routing_review_required");
    }
  }

  return result;
}

// RUN ENGINE
const output = companies.map(evaluateCompany);

// OUTPUT
console.log(JSON.stringify(output, null, 2));

4️⃣ README.md
Natoli Truth Screening Engine

Purpose:
Binary screening of companies using Natoli's compaction process truths.

Rules:
- Every truth must pass
- Any failure results in rejection with explicit explanation
- Qualified companies are routed to Engineering or Scientific

Why:
Natoli sells to process reality, not industry labels.
