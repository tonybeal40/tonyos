STEP 2: News fetcher (100% working code)
/server/news.js
import fetch from "node-fetch";

const TRUSTED_SOURCES = [
  "reuters.com",
  "bloomberg.com",
  "wsj.com",
  "fda.gov",
  "ema.europa.eu",
  "contractpharma.com",
  "nutraceuticalsworld.com"
];

export async function fetchTopFactualNews(companyName) {
  const url = `https://newsapi.org/v2/everything?q="${encodeURIComponent(
    companyName
  )}"&language=en&sortBy=publishedAt&pageSize=20&apiKey=${process.env.NEWS_API_KEY}`;

  const res = await fetch(url);
  const data = await res.json();

  if (!data.articles) return [];

  const last30Days = Date.now() - 30 * 24 * 60 * 60 * 1000;

  const filtered = data.articles
    .filter(a => {
      if (!a.url || !a.publishedAt) return false;
      const published = new Date(a.publishedAt).getTime();
      if (published < last30Days) return false;

      return TRUSTED_SOURCES.some(source =>
        a.url.toLowerCase().includes(source)
      );
    })
    .slice(0, 5)
    .map(a => ({
      title: a.title,
      source: a.source?.name || "Unknown",
      url: a.url,
      published_at: a.publishedAt,
      summary: a.description
    }));

  return filtered;
}

STEP 3: Store it in Tony CRM
/server/intel.js
import db from "./db.js";

export function saveNews(accountId, articles) {
  const stmt = db.prepare(`
    INSERT INTO news_intel
    (account_id, title, source, url, published_at, summary)
    VALUES (?, ?, ?, ?, ?, ?)
  `);

  articles.forEach(a => {
    stmt.run(
      accountId,
      a.title,
      a.source,
      a.url,
      a.published_at,
      a.summary
    );
  });
}
