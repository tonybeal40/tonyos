import requests
import pandas as pd
import time

SERP_API_KEY = "YOUR_SERPAPI_KEY"

KEYWORDS = [
    "chlorine tablets manufacturer",
    "water treatment tablets manufacturing",
    "pool chlorine tablets manufacturer",
    "disinfectant tablets manufacturer",
    "effervescent chlorine tablets manufacturer"
]

results = []

def search_google(query):
    url = "https://serpapi.com/search"
    params = {
        "engine": "google",
        "q": query,
        "api_key": SERP_API_KEY,
        "num": 10
    }
    response = requests.get(url, params=params)
    return response.json()

def manufacturing_confidence(text):
    text = text.lower()
    score = 0
    keywords = [
        "manufacturer",
        "manufacturing",
        "production facility",
        "in-house production",
        "tablet manufacturing",
        "chemical manufacturing"
    ]
    for k in keywords:
        if k in text:
            score += 1
    return score

for keyword in KEYWORDS:
    data = search_google(keyword)
    for r in data.get("organic_results", []):
        title = r.get("title", "")
        snippet = r.get("snippet", "")
        link = r.get("link", "")

        confidence = manufacturing_confidence(title + " " + snippet)

        results.append({
            "Company Name": title.split("–")[0].split("-")[0].strip(),
            "Website": link,
            "Search Keyword": keyword,
            "Snippet": snippet,
            "Manufacturing Confidence Score": confidence
        })

    time.sleep(1)

df = pd.DataFrame(results)
df = df.sort_values("Manufacturing Confidence Score", ascending=False)
df.to_csv("tablet_manufacturers.csv", index=False)

print("Saved tablet_manufacturers.csv")
