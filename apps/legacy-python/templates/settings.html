Connect Google Sheets

Your data automatically syncs to Google Sheets whenever you save in any app. Follow these steps to set it up.

  1. Create a new Google Sheet
  2. Go to Extensions → Apps Script
  3. Delete any existing code and paste the code from the "Apps Script Code" tab
  4. Click Deploy → New deployment
  5. Select Web app as the type
  6. Set "Execute as" to Me
  7. Set "Who has access" to Anyone
  8. Click Deploy and copy the Web App URL
  9. Click the sync indicator (bottom-right of any page) and paste your URL + API key

What Gets Synced

Each app syncs to its own tab (worksheet) in your master Google Sheet:

  • Journal - Date, mood, content, gratitude, focus, reflection, tags
  • CRM_Contacts - Name, company, title, email, phone, LinkedIn, stage
  • CRM_Activity - Contact, channel, type, outcome, notes, date
  • Budget - Type, category, description, amount, date
  • Daily - One win, pressure, next step, result
  • Jobs - Company, title, status, applied date, notes, URL

Google Apps Script Code

Copy this entire code and paste it into your Google Apps Script editor. Change the API_TOKEN to your own secret.

// TonyOS Master Sync - Google Apps Script // Paste this in Extensions > Apps Script const API_TOKEN = "your-secret-token-here"; // CHANGE THIS! function doPost(e) { try { const body = JSON.parse(e.postData.contents || "{}"); if (body.token !== API_TOKEN) { return jsonOut({ ok: false, error: "Unauthorized" }); } const app = String(body.app || "").toLowerCase(); const payload = body.payload || {}; const sheetNames = { journal: "Journal", crm_contacts: "CRM_Contacts", crm_activity: "CRM_Activity", budget: "Budget", daily: "Daily", jobs: "Jobs" }; const sheetName = sheetNames[app]; if (!sheetName) { return jsonOut({ ok: false, error: "Unknown app: " + app }); } const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); const headers = getHeadersForApp(app); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold"); } const row = getRowForApp(app, payload); sheet.appendRow(row); return jsonOut({ ok: true }); } catch (err) { return jsonOut({ ok: false, error: String(err.message || err) }); } } function doGet(e) { const p = e.parameter || {}; if (p.token !== API_TOKEN) { return jsonOut({ ok: false, error: "Unauthorized" }); } return jsonOut({ ok: true, message: "TonyOS Sync is ready" }); } function jsonOut(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getHeadersForApp(app) { const headers = { journal: ["Timestamp", "Date", "Mood", "Stress", "Category", "Content", "Gratitude", "Focus", "Reflection", "Tags", "Links", "WordCount", "UpdatedAt"], crm_contacts: ["Timestamp", "ID", "Name", "Company", "Title", "Email", "Phone", "LinkedIn", "Website", "Notes", "Stage", "Source", "CreatedAt", "UpdatedAt"], crm_activity: ["Timestamp", "ID", "ContactID", "ContactName", "Channel", "Type", "Outcome", "Notes", "Date"], budget: ["Timestamp", "ID", "Type", "Category", "Description", "Amount", "Date", "Recurring"], daily: ["Timestamp", "Date", "OneWin", "Pressure", "NextStep", "Result"], jobs: ["Timestamp", "ID", "Company", "Title", "Status", "Applied", "Notes", "URL", "UpdatedAt"] }; return headers[app] || ["Timestamp", "Data"]; } function getRowForApp(app, p) { const ts = new Date().toISOString(); const rows = { journal: [ts, p.date, p.mood, p.stress, p.category, p.content, p.gratitude, p.focus, p.reflection, p.tags, p.links, p.wordCount, p.updatedAt], crm_contacts: [ts, p.id, p.name, p.company, p.title, p.email, p.phone, p.linkedin, p.website, p.notes, p.stage, p.source, p.createdAt, p.updatedAt], crm_activity: [ts, p.id, p.contactId, p.contactName, p.channel, p.type, p.outcome, p.notes, p.date], budget: [ts, p.id, p.type, p.category, p.description, p.amount, p.date, p.recurring], daily: [ts, p.date, p.oneWin, p.pressure, p.nextStep, p.result], jobs: [ts, p.id, p.company, p.title, p.status, p.applied, p.notes, p.url, p.updatedAt] }; return rows[app] || [ts, JSON.stringify(p)]; }
