import requests
import pandas as pd
import time
import logging
from bs4 import BeautifulSoup
from urllib.parse import urlparse, urljoin
from urllib.robotparser import RobotFileParser

# ===============================
# CONFIG
# ===============================

BING_API_KEY = "YOUR_BING_API_KEY"
BING_ENDPOINT = "https://api.bing.microsoft.com/v7.0/search"

USER_AGENT = "NatoliMarketResearchBot/1.0 (+public-research)"
REQUEST_DELAY = 2.0   # conservative
MAX_PAGES_PER_SITE = 2  # homepage + about only

OUTPUT_FILE = "natoli_us_targets.csv"
LOG_FILE = "natoli_compliance.log"

HEADERS = {
    "Ocp-Apim-Subscription-Key": BING_API_KEY,
    "User-Agent": USER_AGENT
}

# ===============================
# LOGGING (COMPLIANCE)
# ===============================

logging.basicConfig(
    filename=LOG_FILE,
    level=logging.INFO,
    format="%(asctime)s | %(levelname)s | %(message)s"
)

logging.info("Natoli legal discovery pipeline started")

# ===============================
# INDUSTRY QUERIES
# ===============================

ENGINEERING_INDUSTRIES = [
    "pharmaceutical tablet manufacturing USA",
    "generic drug tablet manufacturing USA",
    "503b non sterile tablet manufacturing USA",
    "nutraceutical tablet manufacturing USA",
    "chemical tablet manufacturing USA",
    "chlorine tablet manufacturing USA",
    "water treatment tablets manufacturing USA"
]

SCIENTIFIC_INDUSTRIES = [
    "pharmaceutical formulation development USA",
    "compaction characterization laboratory USA",
    "manufacturing science technology pharma USA",
    "tablet scale up laboratory USA"
]

ALL_INDUSTRIES = ENGINEERING_INDUSTRIES + SCIENTIFIC_INDUSTRIES

# ===============================
# ROBOTS.TXT CHECK
# ===============================

def robots_allowed(url):
    try:
        parsed = urlparse(url)
        robots_url = f"{parsed.scheme}://{parsed.netloc}/robots.txt"

        rp = RobotFileParser()
        rp.set_url(robots_url)
        rp.read()

        allowed = rp.can_fetch(USER_AGENT, url)
        logging.info(f"ROBOTS CHECK | {url} | allowed={allowed}")
        return allowed
    except Exception as e:
        logging.warning(f"ROBOTS CHECK FAILED | {url} | {e}")
        return False

# ===============================
# WEBSITE TEXT FETCH (SAFE)
# ===============================

def fetch_public_text(base_url):
    collected_text = ""

    try:
        pages = [base_url, urljoin(base_url, "/about")]

        for page in pages[:MAX_PAGES_PER_SITE]:
            if not robots_allowed(page):
                logging.info(f"SKIPPED BY ROBOTS | {page}")
                continue

            r = requests.get(
                page,
                headers={"User-Agent": USER_AGENT},
                timeout=6
            )

            if r.status_code != 200:
                continue

            soup = BeautifulSoup(r.text, "html.parser")
            text = soup.get_text(" ", strip=True)
            collected_text += " " + text[:4000]

            time.sleep(REQUEST_DELAY)

        return collected_text.strip()

    except Exception as e:
        logging.error(f"FETCH ERROR | {base_url} | {e}")
        return ""

# ===============================
# NATOLI FIT CLASSIFIER
# ===============================

def natoli_fit_classifier(text):
    if not text:
        return "No Fit", "No public manufacturing signal"

    t = text.lower()

    disqualifiers = [
        "injectable only", "infusion", "admixture",
        "clinic", "wellness", "trt", "pellet"
    ]
    if any(d in t for d in disqualifiers):
        return "No Fit", "Sterile or clinical focus"

    engineering = any(k in t for k in [
        "tablet", "capsule", "solid oral",
        "manufacturing", "press", "production"
    ])

    scientific = any(k in t for k in [
        "formulation", "ms&t", "scale-up",
        "compaction", "analytical"
    ])

    if engineering and scientific:
        return "Both", "Production compression with development activity"
    if engineering:
        return "Engineering", "Tablet manufacturing and press operations"
    if scientific:
        return "Scientific", "Formulation or compaction science focus"

    return "No Fit", "No compression relevance"

# ===============================
# SEARCH FUNCTION (BING API)
# ===============================

def bing_search(query):
    params = {"q": query, "count": 8}
    r = requests.get(BING_ENDPOINT, headers=HEADERS, params=params)
    r.raise_for_status()
    return r.json().get("webPages", {}).get("value", [])

# ===============================
# MAIN PIPELINE
# ===============================

rows = []

for industry in ALL_INDUSTRIES:
    logging.info(f"SEARCH INDUSTRY | {industry}")

    results = bing_search(industry)

    for r in results:
        company = r.get("name", "")
        website = r.get("url", "")
        snippet = r.get("snippet", "")

        logging.info(f"DISCOVERED | {company} | {website}")

        website_text = fetch_public_text(website)
        combined_text = snippet + " " + website_text

        fit, reason = natoli_fit_classifier(combined_text)

        rows.append({
            "Company": company,
            "City": "",
            "State": "",
            "Phone": "",
            "Contact": "",
            "Website": website,
            "LinkedIn": "",
            "Natoli Fit": fit,
            "Reason": reason,
            "Industry": industry,
            "Source": urlparse(website).netloc
        })

        time.sleep(REQUEST_DELAY)

df = pd.DataFrame(rows)
df.drop_duplicates(subset=["Company", "Website"], inplace=True)
df.to_csv(OUTPUT_FILE, index=False)

logging.info("Pipeline completed successfully")
print(f"CSV created: {OUTPUT_FILE}")
