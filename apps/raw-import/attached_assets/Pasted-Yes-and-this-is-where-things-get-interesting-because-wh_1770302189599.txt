Minimal TonyOS LLM core (Replit-ready)
tonyLLM.js
import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

export async function tonyBrain(input) {

  const systemPrompt = `
You are TonyOS Manufacturing Intelligence Engine.

Think like a Natoli process expert.

Evaluate:

- compression relevance
- scale pressure
- tooling risk
- validation exposure
- competitor gravity
- silence signals

Return structured JSON:

{
  "compression": "",
  "pressure": [],
  "risk": [],
  "competitor": [],
  "decision": ""
}
`;

  const res = await client.chat.completions.create({
    model: "gpt-4.1-mini",
    messages: [
      { role: "system", content: systemPrompt },
      { role: "user", content: input }
    ],
    temperature: 0
  });

  return JSON.parse(res.choices[0].message.content);
}
