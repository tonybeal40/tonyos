export const ATTENTION = {
  IGNORE: "ignore",
  MONITOR: "monitor",
  ACT_NOW: "act_now"
};

export function classifyAttention(signalType) {
  const urgent = ["HUMAN_REVIEW", "DECISION_FAILED"];
  const monitor = ["DECISION_COMPLETE", "RESEARCH_COMPLETE"];

  if (urgent.includes(signalType)) return ATTENTION.ACT_NOW;
  if (monitor.includes(signalType)) return ATTENTION.MONITOR;
  return ATTENTION.IGNORE;
}

4️⃣ Anti-Momentum Guard (prevents streak mistakes)
momentum.js
let recentActions = [];

export function momentumCheck(action, limit = 3) {
  recentActions.push({ action, time: Date.now() });
  recentActions = recentActions.filter(
    a => Date.now() - a.time < 600000 // 10 min
  );

  const count = recentActions.filter(a => a.action === action).length;

  return {
    allowed: count <= limit,
    count
  };
}


If you do the same thing too many times → pause required
This saves people from themselves.

7️⃣ One Smart Pre-Action Gate (ties it together)
preActionGate.js
import { momentumCheck } from "./momentum.js";
import { identityCheck } from "./identity.js";

export function preActionGate({ action, alignsWithIdentity }) {
  const momentum = momentumCheck(action);

  if (!momentum.allowed) {
    return {
      allowed: false,
      reason: "Momentum limit reached"
    };
  }

  const identity = identityCheck({ action, aligns: alignsWithIdentity });

  if (!identity.alignsWithIdentity) {
    return {
      allowed: false,
      reason: "Identity misalignment"
    };
  }

  return { allowed: true };
}


This is your brainstem.
