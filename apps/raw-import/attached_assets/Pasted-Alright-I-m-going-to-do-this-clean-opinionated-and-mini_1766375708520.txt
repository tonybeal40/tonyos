1️⃣ State (non-negotiable)
state.js
export const STATES = {
  IDLE: "idle",
  RUNNING: "running",
  WAITING: "waiting",
  NEEDS_REVIEW: "needs_review",
  COMPLETE: "complete",
  FAILED: "failed"
};

export function newState(name, meta = {}) {
  return {
    name,
    state: STATES.IDLE,
    meta,
    log: [],
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString()
  };
}

export function setState(record, state, note = "") {
  record.state = state;
  record.updatedAt = new Date().toISOString();
  if (note) {
    record.log.push({
      time: record.updatedAt,
      note
    });
  }
  return record;
}

7️⃣ One Job Runner (no magic)
job.js
import { STATES, newState, setState } from "./state.js";
import { signal } from "./signals.js";
import { needsHuman } from "./humanGate.js";

export async function runJob(name, action, fn) {
  const job = newState(name);
  setState(job, STATES.RUNNING, "Started");

  if (needsHuman(action)) {
    setState(job, STATES.NEEDS_REVIEW, "Waiting for human");
    signal("HUMAN_REQUIRED", { action, name });
    return job;
  }

  try {
    const result = await fn();
    setState(job, STATES.COMPLETE, "Completed");
    return { ...job, result };
  } catch (e) {
    setState(job, STATES.FAILED, e.message);
    signal("FAILED", { name, error: e.message });
    return job;
  }
}


No queues

5️⃣ CRM Job Runner (decision engine)
crmJob.js
import { RELATIONSHIP_STATES, setRelationshipState } from "./crmState.js";
import { emitSignal } from "./crmSignals.js";
import { requiresHumanApproval } from "./crmGate.js";

export async function runCRMDecision(record, action, fn) {
  if (requiresHumanApproval(action)) {
    setRelationshipState(record, RELATIONSHIP_STATES.NEEDS_REVIEW, "Human approval required");
    emitSignal("HUMAN_REVIEW", { action, account: record.account });
    return record;
  }

  try {
    const result = await fn();
    setRelationshipState(record, RELATIONSHIP_STATES.READY, "Decision complete");
    emitSignal("DECISION_COMPLETE", { action, account: record.account });
    return { record, result };
  } catch (e) {
    setRelationshipState(record, RELATIONSHIP_STATES.NO_ACTION, e.message);
    emitSignal("DECISION_FAILED", { account: record.account });
    return record;
  }
}
