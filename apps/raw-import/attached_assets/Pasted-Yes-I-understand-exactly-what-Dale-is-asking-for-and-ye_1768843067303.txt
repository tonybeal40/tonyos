    elif customer_type == "existing":

CODE #4: Template Generator (EOD-ready)

This generates both documents automatically.

import json

with open("natoli_contacts.json") as f:
    contacts = json.load(f)["team"]

with open("natoli_sales_regions.json") as f:
    regions = json.load(f)["Domestic"]

def generate_template(customer_type):
    narrative = intro_narrative(customer_type)

    contact_block = "\n".join(
        [f"{c['name']} – {c['title']} | {c['email']}" for c in contacts]
    )

    sales_block = "\n".join(
        [f"{r['region']}: {r['name']} | {r['email']} | {r['phone']}" for r in regions]
    )

    return f"""
{narrative}

Who You’ll Work With:
{contact_block}

Your Regional Contact:
{sales_block}
"""

# Generate both
new_customer_doc = generate_template("new")
existing_customer_doc = generate_template("existing")

print(new_customer_doc)
print(existing_customer_doc)


This gives you two clean drafts instantly.
