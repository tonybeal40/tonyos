1️⃣ Accounts model + routes
/server/accounts.js
import db from "./db.js";

/* CREATE ACCOUNT */
export function createAccount(data) {
  const stmt = db.prepare(`
    INSERT INTO accounts (
      company_name,
      domain,
      website,
      linkedin_url,
      industry,
      naics,
      sic,
      employees,
      revenue,
      division_fit,
      persona_score,
      tier,
      pain_points,
      why_natoli,
      status
    ) VALUES (
      @company_name,
      @domain,
      @website,
      @linkedin_url,
      @industry,
      @naics,
      @sic,
      @employees,
      @revenue,
      @division_fit,
      @persona_score,
      @tier,
      @pain_points,
      @why_natoli,
      @status
    )
  `);

  return stmt.run(data);
}

/* GET ALL ACCOUNTS */
export function getAllAccounts() {
  return db.prepare(`
    SELECT * FROM accounts
    ORDER BY created_at DESC
  `).all();
}

/* GET SINGLE ACCOUNT */
export function getAccountById(id) {
  return db.prepare(`
    SELECT * FROM accounts WHERE id = ?
  `).get(id);
}

/* UPDATE STATUS */
export function updateAccountStatus(id, status) {
  return db.prepare(`
    UPDATE accounts
    SET status = ?, updated_at = CURRENT_TIMESTAMP
    WHERE id = ?
  `).run(status, id);
}

import db from "./db.js";

/* ADD CONTACT */
export function addContact(data) {
  const stmt = db.prepare(`
    INSERT INTO contacts (
      account_id,
      name,
      title,
      linkedin_url,
      persona_type
    ) VALUES (
      @account_id,
      @name,
      @title,
      @linkedin_url,
      @persona_type
    )
  `);

  return stmt.run(data);
}

/* GET CONTACTS FOR ACCOUNT */
export function getContactsByAccount(accountId) {
  return db.prepare(`
    SELECT * FROM contacts WHERE account_id = ?
  `).all(accountId);
}

import db from "./db.js";

/* ADD NOTE */
export function addNote(accountId, note) {
  return db.prepare(`
    INSERT INTO notes (account_id, note)
    VALUES (?, ?)
  `).run(accountId, note);
}

/* GET NOTES FOR ACCOUNT */
export function getNotesByAccount(accountId) {
  return db.prepare(`
    SELECT * FROM notes
    WHERE account_id = ?
    ORDER BY created_at DESC
  `).all(accountId);

4️⃣ Wire everything into the server

Now replace your /server/index.js with this complete version.

/server/index.js
import express from "express";
import db from "./db.js";

import {
  createAccount,
  getAllAccounts,
  getAccountById,
  updateAccountStatus
} from "./accounts.js";

import {
  addContact,
  getContactsByAccount
} from "./contacts.js";

import {
  addNote,
  getNotesByAccount
} from "./notes.js";

const app = express();
app.use(express.json());

/* HEALTH */
app.get("/health", (req, res) => {
  res.json({ status: "Tony CRM online" });
});

/* ACCOUNTS */
app.post("/accounts", (req, res) => {
  try {
    const result = createAccount(req.body);
    res.json({ success: true, id: result.lastInsertRowid });
  } catch (e) {
    res.status(400).json({ error: e.message });
  }
});

app.get("/accounts", (req, res) => {
  res.json(getAllAccounts());
});

app.get("/accounts/:id", (req, res) => {
  const account = getAccountById(req.params.id);
  if (!account) return res.status(404).json({ error: "Not found" });

  res.json({
    account,
    contacts: getContactsByAccount(req.params.id),
    notes: getNotesByAccount(req.params.id)
  });
});

app.patch("/accounts/:id/status", (req, res) => {
  updateAccountStatus(req.params.id, req.body.status);
  res.json({ success: true });
});

/* CONTACTS */
app.post("/contacts", (req, res) => {
  addContact(req.body);
  res.json({ success: true });
});

/* NOTES */
app.post("/notes", (req, res) => {
  addNote(req.body.account_id, req.body.note);
  res.json({ success: true });
});

app.listen(3000, () => {
  console.log("Tony CRM running on port 3000");
});
