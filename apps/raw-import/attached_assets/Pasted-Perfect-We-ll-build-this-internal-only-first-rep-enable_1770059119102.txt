3️⃣ app.py (Backend Logic)
import os
import pandas as pd
import openai
from flask import Flask, request, render_template, send_file
from io import BytesIO

openai.api_key = os.getenv("OPENAI_API_KEY")

app = Flask(__name__)

SYSTEM_PROMPT = """
You are a pharmaceutical manufacturing reasoning assistant.
Do NOT invent specifications.
Use probabilistic language only.
Infer likely tooling class, press OEM exposure, production stage,
probable tablet defects, and a safe Natoli discussion angle.
"""

def enrich_row(row):
    prompt = f"""
Company: {row.get('Company')}
Title: {row.get('Title')}
Function: {row.get('Function')}
Industry: Pharmaceutical manufacturing

Return JSON with:
- Tooling_Class (Likely)
- Press_OEM (Likely)
- Production_Stage
- Probable_Defects
- Natoli_Angle
- Email_Subject
- LinkedIn_Opening
"""

    response = openai.ChatCompletion.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": prompt}
        ],
        temperature=0.2
    )

    return response.choices[0].message.content

@app.route("/", methods=["GET", "POST"])
def index():
    if request.method == "POST":
        file = request.files["file"]
        df = pd.read_csv(file)

        enriched_data = []

        for _, row in df.iterrows():
            try:
                ai_output = enrich_row(row)
            except Exception as e:
                ai_output = "ERROR"

            enriched_row = row.to_dict()
            enriched_row["AI_Insight"] = ai_output
            enriched_data.append(enriched_row)

        output_df = pd.DataFrame(enriched_data)

        output = BytesIO()
        output_df.to_csv(output, index=False)
        output.seek(0)

        return send_file(
            output,
            mimetype="text/csv",
            as_attachment=True,
            download_name="natoli_enriched_list.csv"
        )

    return render_template("index.html")

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=3000)
