5️⃣ CSV processing logic (app.js)

This is the critical part that makes it “match up correctly”.

import { ENGINEERING_KEYWORDS, SCIENTIFIC_KEYWORDS, EXCLUSIONS } from "../data/rules.js"
import { TERRITORY_MAP } from "../data/territoryMap.js"

function normalize(text = "") {
  return text.toLowerCase().trim()
}

function containsAny(text, list) {
  return list.some(keyword => text.includes(keyword))
}

export function classifyRow(row) {
  const description = normalize(
    `${row.company || ""} ${row.industry || ""} ${row.description || ""}`
  )

  // 1. Exclusions
  if (containsAny(description, EXCLUSIONS)) {
    return { status: "Excluded" }
  }

  // 2. Business unit
  let division = "Needs Review"

  if (containsAny(description, ENGINEERING_KEYWORDS)) {
    division = "Natoli Engineering"
  }

  if (
    containsAny(description, SCIENTIFIC_KEYWORDS) &&
    division !== "Natoli Engineering"
  ) {
    division = "Natoli Scientific"
  }

  // 3. Territory
  const state = normalize(row.state || "").toUpperCase()
  const territory = TERRITORY_MAP[state] || "Unknown"

  return {
    division,
    territory,
    status: "Qualified"
  }
}
