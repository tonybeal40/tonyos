2️⃣ truth_engine.js

This is the actual enforcement engine

👉 Paste exactly

const fs = require("fs");

const truths = JSON.parse(fs.readFileSync("natoli_truths.json", "utf8"));
const companies = JSON.parse(fs.readFileSync("input_companies.json", "utf8"));

function normalize(text) {
  return text.toLowerCase();
}

function hasAny(text, list = []) {
  return list.some(k => text.includes(k));
}

function evaluateCompany(company) {
  const text = normalize(
    `${company.name} ${company.industry} ${company.description}`
  );

  const result = {
    name: company.name,
    qualified: true,
    failed_truths: [],
    notes: []
  };

  // GLOBAL BLOCK
  if (hasAny(text, truths.global_blocked_keywords)) {
    result.qualified = false;
    result.notes.push("Hard excluded by global blocked keyword");
    return result;
  }

  // TRUTH EVALUATION
  for (const truth of truths.truths) {
    if (truth.required_keywords && !hasAny(text, truth.required_keywords)) {
      result.qualified = false;
      result.failed_truths.push(truth.id);
    }

    if (truth.blocked_keywords && hasAny(text, truth.blocked_keywords)) {
      result.qualified = false;
      result.failed_truths.push(truth.id);
    }
  }

  return result;
}

// RUN ENGINE
const output = companies.map(evaluateCompany);

// PRINT RESULTS
console.log(JSON.stringify(output, null, 2));

Natoli Truth Engine

Purpose:
Binary screening of companies based on Natoli compaction process truths.

Logic:
Every truth must evaluate to TRUE.
If any truth fails, the company is rejected.

Why:
Natoli sells to process reality, not industry labels.
