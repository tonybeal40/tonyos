function loadModule(name) {

function saveState(key, value) {
  localStorage.setItem(key, JSON.stringify(value));
}

function loadState(key) {
  return JSON.parse(localStorage.getItem(key));

window.modules.csv = function () {

  return `
    <h1>CSV Tools</h1>

    <input type="file" id="csvFile">

    <div id="csvOutput"></div>

    <script>
      document.getElementById("csvFile").addEventListener("change", function(e){

        let file = e.target.files[0];
        let reader = new FileReader();

        reader.onload = function(){
          let text = reader.result;
          let rows = text.split("\\n").map(r => r.split(","));

          let html = "<table>";
          rows.forEach(row=>{
            html += "<tr>" + row.map(c=>"<td>"+c+"</td>").join("") + "</tr>";
          });
          html += "</table>";

          document.getElementById("csvOutput").innerHTML = html;
        };

        reader.readAsText(file);
      });
    </script>
  `;
};
modules/workflows.js
(Guided workflow engine)

const workflowSteps = [
  "Vendor Identified",
  "Compliance Review",
  "Quote Approved",
  "PO Issued",
  "Vendor Activated"
];

window.modules.workflows = function () {

  let progress = loadState("workflow-progress") || {};
  let percent = progress.vendor || 0;

  function update(step){
    percent = ((step+1)/workflowSteps.length)*100;
    progress.vendor = percent;
    saveState("workflow-progress", progress);
    loadModule("workflows");
  }

  return `
    <h1>Vendor Approval Workflow</h1>

    ${workflowSteps.map((s,i)=>`
      <div class="card">
        ${s}
        <button onclick="(${update})(${i})">Complete</button>
      </div>
    `).join("")}

    <h3>Progress: ${percent}%</h3>
  `;
};

📊 modules/workflows.js
(Multi-workflow manager)

const workflowData = {
vendorApproval:[
"Vendor Identified",
"Compliance Review",
"Quote Approved",
"PO Issued",
"Vendor Activated"
]
};

window.modules.workflows = function(){

let progress = load("workflow") || {};

function complete(step){

progress.vendor = step;
save("workflow", progress);
loadModule("workflows");
}

let current = progress.vendor || 0;

return `
<h1>Workflow Manager</h1>

${workflowData.vendorApproval.map((s,i)=>`

<div class="card">

${i <= current ? "✔" : "⬜"} ${s}

${i===current ?

`<button onclick="(${complete})(${i+1})">
Complete Step
</button>` : ""}

</div>

`).join("")}

<h3>Progress: ${Math.round(current /
workflowData.vendorApproval.length * 100)}%</h3>
`;
};

📈 csvAuto.js
(auto vendor scoring from CSV)

window.modules.csvAuto = function(){

return `
<h1>CSV Vendor Scoring</h1>

<input type="file" id="csv">

<div id="output"></div>

<script>

document.getElementById("csv").onchange = e=>{

let file=e.target.files[0];
let r=new FileReader();

r.onload=()=>{

let rows=r.result.split("\\n")
.map(x=>x.split(","));

let scored=rows.map(r=>{

let score=
r[1]*0.4 +
r[2]*0.3 +
r[3]*0.3;

return [...r, score.toFixed(2)];

});

let html="<table>";

scored.forEach(row=>{
html+="<tr>"+row.map(c=>"<td>"+c+"</td>").join("")+"</tr>";
});

html+="</table>";

document.getElementById("output").innerHTML=html;

};

r.readAsText(file);
};

</script>
`;
};

Vendor,Delivery,Price,Compliance
