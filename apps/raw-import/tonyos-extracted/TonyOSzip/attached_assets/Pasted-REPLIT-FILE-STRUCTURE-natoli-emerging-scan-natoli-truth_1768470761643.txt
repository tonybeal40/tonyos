const axios = require("axios");
const cheerio = require("cheerio");
const fs = require("fs");

const truths = JSON.parse(fs.readFileSync("natoli_truths.json", "utf8"));
const seeds = JSON.parse(fs.readFileSync("search_seeds.json", "utf8"));

function normalize(text) {
  return text.replace(/\s+/g, " ").toLowerCase();
}

function hasAny(text, list = []) {
  return list.some(k => text.includes(k));
}

async function duckDuckGoSearch(query) {
  const url = `https://duckduckgo.com/html/?q=${encodeURIComponent(query)}`;
  const res = await axios.get(url, { timeout: 10000 });
  const $ = cheerio.load(res.data);
  const links = [];

  $("a.result__a").each((_, el) => {
    const href = $(el).attr("href");
    if (href && href.startsWith("http")) {
      links.push(href.split("&")[0]);
    }
  });

  return [...new Set(links)];
}

async function scrapeSite(url) {
  try {
    const res = await axios.get(url, { timeout: 10000 });
    const $ = cheerio.load(res.data);
    return normalize($("body").text());
  } catch {
    return "";
  }
}

function evaluateAgainstTruths(text) {
  const failed = [];
  for (const truth of truths.truths) {
    if (truth.required_keywords && !hasAny(text, truth.required_keywords)) {
      failed.push(truth.rejection_sentence);
    }
    if (truth.blocked_keywords && hasAny(text, truth.blocked_keywords)) {
      failed.push(truth.rejection_sentence);
    }
  }
  return {
    qualified: failed.length === 0,
    failed_reasons: failed
  };
}

(async () => {
  const discovered = {};

  for (const query of seeds.queries) {
    console.log(`Searching: ${query}`);
    const results = await duckDuckGoSearch(query);

    for (const link of results) {
      if (discovered[link]) continue;

      const text = await scrapeSite(link);
      if (!text) continue;

      // HARD BLOCK
      if (hasAny(text, truths.global_blocked_keywords)) continue;

      const evalResult = evaluateAgainstTruths(text);

      discovered[link] = {
        url: link,
        qualified: evalResult.qualified,
        failed_reasons: evalResult.failed_reasons
      };
    }
  }

  const output = Object.values(discovered)
    .filter(c => c.qualified)
    .map(c => ({
      url: c.url,
      reason: "Early-stage compaction + formulation signals detected"
    }));

  fs.writeFileSync(
    "emerging_natoli_companies.json",
    JSON.stringify(output, null, 2)
  );

  console.log("Scan complete. Emerging companies written to emerging_natoli_companies.json");
})();
