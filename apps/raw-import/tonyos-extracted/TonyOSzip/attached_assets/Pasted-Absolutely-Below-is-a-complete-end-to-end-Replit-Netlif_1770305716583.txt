src/engine/format.ts
export function money(n: number) {
  if (!Number.isFinite(n)) return "$0";
  return n.toLocaleString(undefined, { style: "currency", currency: "USD", maximumFractionDigits: 0 });
}

export function clamp(n: number, min: number, max: number) {
  return Math.max(min, Math.min(max, n));
}

export function safeNum(v: any) {
  const n = Number(v);
  return Number.isFinite(n) ? n : 0;
}

export function toDate(v: any): Date | undefined {
  if (!v) return undefined;
  if (v instanceof Date && !isNaN(v.getTime())) return v;

  // Excel sometimes gives strings like "00:00:00"
  if (typeof v === "string") {
    const s = v.trim();
    if (s === "" || s === "00:00:00") return undefined;
    const d = new Date(s);
    if (!isNaN(d.getTime())) return d;
    return undefined;
  }
  if (typeof v === "number") {
    // Excel serial date fallback: 25569 days between 1899-12-30 and 1970-01-01
    const excelEpoch = new Date(Date.UTC(1899, 11, 30));
    const d = new Date(excelEpoch.getTime() + v * 86400000);
    if (!isNaN(d.getTime())) return d;
  }
  return undefined;
}

export function daysBetween(from?: Date, to = new Date()): number {
  if (!from) return 365; // unknown => treat as dormant/high urgency
  const ms = to.getTime() - from.getTime();
  return Math.max(0, Math.round(ms / 86400000));
}

src/engine/normalize.ts
import { InventoryItem, InventoryRawRow, InventoryType } from "../types";
import { daysBetween, safeNum, toDate } from "./format";
import { scoreItem } from "./scoring";

/**
 * Normalizes both ACC and HAGLIN exports (Excel/CSV) into a unified schema.
 * Uses best-available date signal:
 * 1) Date last stock movement
 * 2) Last issue date
 * 3) Last receipt date
 * 4) Dormant/Unknown (treated as high urgency)
 */
export function normalizeRows(rows: InventoryRawRow[], forcedType?: InventoryType): InventoryItem[] {
  const now = new Date();

  return rows
    .map((r, idx) => {
      const sku = String(r["Stock code"] ?? r["SKU"] ?? r["sku"] ?? `ROW-${idx}`).trim();
      const description = String(r["Description"] ?? r["description"] ?? "").trim();

      const quantity = safeNum(r["Quantity on hand"] ?? r["Qty"] ?? r["quantity"] ?? 0);

      // Costs differ between sheets
      const unitCost =
        safeNum(r["Current inventory unit cost"] ?? r["Current Unit Cost"] ?? r["unitCost"] ?? r["Unit Cost"] ?? 0);

      const extendedValue =
        safeNum(r["Total Inventory Value"] ?? r["Extended Value"] ?? r["extendedValue"] ?? quantity * unitCost);

      // Detect type if not forced
      let type: InventoryType = forcedType ?? "ACC";
      if (!forcedType) {
        const bin = String(r["Bin"] ?? "").toUpperCase();
        if (bin.includes("HAGLIN") || sku.startsWith("NP") || sku.startsWith("NCF") || sku.startsWith("RD")) type = "HAGLIN";
      }

      const dMove = toDate(r["Date last stock movement"]);
      const dIssue = toDate(r["Last issue date"]);
      const dReceipt = toDate(r["Last receipt date"]);
      const lastSignalDate = dMove ?? dIssue ?? dReceipt;

      const daysIdle = daysBetween(lastSignalDate, now);

      const base = {
        id: `${type}-${sku}-${idx}`,
        sku,
        description,
        type,
        quantity,
        unitCost,
        extendedValue,
        lastSignalDate,
        daysIdle
      };

      return scoreItem(base);
    })
    .filter((x) => x.sku && x.sku !== "nan");
}

src/engine/scoring.ts
import { InventoryItem } from "../types";
import { clamp } from "./format";

/**
 * Consulting-style scoring:
 * - Velocity Index (0-100): faster movement = higher
 * - Risk Index (0-100): idle + high value = higher risk
 * - Confidence Index: based on data quality (have dates? costs? description?)
 * - ROI Index: value + velocity + type boost
 * - Move Score: weighted master rank for Top 50
 */
export function scoreItem(
  base: Omit<InventoryItem, "roiIndex"|"velocityIndex"|"riskIndex"|"confidenceIndex"|"moveScore">
): InventoryItem {
  const value = base.extendedValue;

  const velocityIndex = clamp(100 - Math.min(365, base.daysIdle) * (100 / 365), 0, 100);

  // risk increases with idle days AND value (log scaled)
  const valueFactor = clamp(Math.log10(Math.max(1, value)) * 20, 0, 100);
  const idleFactor = clamp(Math.min(365, base.daysIdle) * (100 / 365), 0, 100);
  const riskIndex = clamp(0.6 * idleFactor + 0.4 * valueFactor, 0, 100);

  const hasDate = !!base.lastSignalDate;
  const hasCost = base.unitCost > 0;
  const hasDesc = base.description.length > 5;
  const confidenceIndex = clamp((hasDate ? 40 : 15) + (hasCost ? 35 : 10) + (hasDesc ? 25 : 10), 0, 100);

  const typeBoost = base.type === "HAGLIN" ? 18 : 10;
  const roiIndex = clamp(0.45 * velocityIndex + 0.35 * valueFactor + typeBoost, 0, 100);

  // Master Move Score (what we rank Top 50 by)
  // prioritize: ROI + Risk + Confidence (actionable + urgent + credible)
  const moveScore = clamp(0.5 * roiIndex + 0.35 * riskIndex + 0.15 * confidenceIndex, 0, 100);

  return { ...base, roiIndex, velocityIndex, riskIndex, confidenceIndex, moveScore };
}

import Papa from "papaparse";
import * as XLSX from "xlsx";
import { normalizeRows } from "../engine/normalize";
import { InventoryItem, InventoryType } from "../types";

export default function UploadPanel({ onLoaded }: { onLoaded: (items: InventoryItem[]) => void }) {
  function parseFile(file: File, forcedType?: InventoryType) {
    const name = file.name.toLowerCase();

    if (name.endsWith(".csv")) {
      Papa.parse(file, {
        header: true,
        skipEmptyLines: true,
        complete: (res) => {
          const rows = (res.data as any[]) || [];
          onLoaded(normalizeRows(rows, forcedType));
        }
      });
      return;
    }

    // xlsx
    const reader = new FileReader();
    reader.onload = (evt) => {
      const data = evt.target?.result;
      const wb = XLSX.read(data, { type: "array" });
      const sheet = wb.Sheets[wb.SheetNames[0]];
      const rows = XLSX.utils.sheet_to_json(sheet, { defval: "" }) as any[];
      onLoaded(normalizeRows(rows, forcedType));
    };
    reader.readAsArrayBuffer(file);
  }

  return (
    <div className="card">
      <h3>Import Inventory (CSV / Excel)</h3>
      <p className="muted">Upload ACC or Haglin exports (or future monthly files). The model normalizes columns automatically.</p>

      <div className="row">
        <label className="pill">
          Upload (Auto Detect)
          <input type="file" accept=".csv,.xlsx,.xls" onChange={(e) => e.target.files && parseFile(e.target.files[0])} hidden />
        </label>

        <label className="pill">
          Upload as ACC
          <input type="file" accept=".csv,.xlsx,.xls" onChange={(e) => e.target.files && parseFile(e.target.files[0], "ACC")} hidden />
        </label>

        <label className="pill">
          Upload as HAGLIN
          <input type="file" accept=".csv,.xlsx,.xls" onChange={(e) => e.target.files && parseFile(e.target.files[0], "HAGLIN")} hidden />
        </label>
      </div>
    </div>
  );
}

import { InventoryItem, Filters } from "../types";
import { money } from "../engine/format";

export default function Top50Table({
  items, filters, selected, toggleSelect, onOpen
}: {
  items: InventoryItem[];
  filters: Filters;
  selected: Set<string>;
  toggleSelect: (id: string) => void;
  onOpen: (item: InventoryItem) => void;
}) {
  const q = filters.query.trim().toLowerCase();

  const filtered = items.filter((x) => {
    if (filters.type !== "ALL" && x.type !== filters.type) return false;
    if (x.moveScore < filters.minMoveScore) return false;
    if (x.daysIdle > filters.maxDaysIdle) return false;
    if (x.extendedValue < filters.minValue) return false;
    if (q && !(x.sku.toLowerCase().includes(q) || x.description.toLowerCase().includes(q))) return false;
    return true;
  });

  const ranked = [...filtered].sort((a, b) => b.moveScore - a.moveScore).slice(0, 50);

import jsPDF from "jspdf";
import html2canvas from "html2canvas";

export default function ExportButtons({ toast }: { toast: (m: string) => void }) {
  async function exportPDF(mode: "Executive" | "Shareholder") {
    toast(`Generating ${mode} PDF…`);
    const el = document.getElementById("export-area");
    if (!el) return;

    const canvas = await html2canvas(el, { scale: 2 });
    const imgData = canvas.toDataURL("image/png");

    const pdf = new jsPDF("p", "pt", "a4");
    const pageW = pdf.internal.pageSize.getWidth();
    const pageH = pdf.internal.pageSize.getHeight();

    const ratio = Math.min(pageW / canvas.width, pageH / canvas.height);
    const w = canvas.width * ratio;
    const h = canvas.height * ratio;

    pdf.text(`Inventory → ROI Command Center (${mode})`, 40, 40);
    pdf.addImage(imgData, "PNG", 40, 60, w - 80, h - 100);
    pdf.save(`Inventory_ROI_${mode}.pdf`);
    toast(`${mode} PDF saved.`);
  }

  function openNetlifyDrop() {
    // "button to cast to Netlify" = opens Netlify Drop deploy flow
    window.open("https://app.netlify.com/drop", "_blank");

  return (
    <div className="card">
      <h3>Export & Deploy</h3>
      <p className="muted">Export board-ready PDFs and deploy on Netlify.</p>

      <div className="row">
        <button className="pill primary" onClick={() => exportPDF("Executive")}>Export Executive PDF</button>

src/App.tsx
import { useMemo, useState } from "react";
import TopNav, { Page } from "./components/TopNav";
import UploadPanel from "./components/UploadPanel";
import FiltersPanel from "./components/FiltersPanel";
import KPIBar from "./components/KPIBar";
import Top50Table from "./components/Top50Table";
import ItemDrawer from "./components/ItemDrawer";
import GoToMarket from "./components/GoToMarket";
import ROISimulator from "./components/ROISimulator";
import FlowDiagram from "./components/FlowDiagram";
import DemoNarration from "./components/DemoNarration";
import ExportButtons from "./components/ExportButtons";
import Toast from "./components/Toast";

import { Filters, InventoryItem, ROIScenario } from "./types";
import { sampleInventory } from "./data/sample";

export default function App() {
  const [page, setPage] = useState<Page>("dashboard");
  const [items, setItems] = useState<InventoryItem[]>(sampleInventory);
  const [drawerItem, setDrawerItem] = useState<InventoryItem | undefined>(undefined);
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
  const [toastMsg, setToastMsg] = useState<string>("");

  const [filters, setFilters] = useState<Filters>({
    type: "ALL",
    minMoveScore: 0,
    maxDaysIdle: 365,
    minValue: 0,
    query: "",
    show: "BOTH"
  });

  const [scenario, setScenario] = useState<ROIScenario>({
    pctTop50Converted: 30,
    attachRate: 15,
    discount: 5,
    horizonDays: 60
  });

  function toast(m: string) {
    setToastMsg(m);
  }

  function toggleSelect(id: string) {
    setSelectedIds((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  }

  const selectedItems = useMemo(
    () => items.filter((x) => selectedIds.has(x.id)),
    [items, selectedIds]
  );

  function onLoaded(newItems: InventoryItem[]) {
    // merge/append behavior: add to existing inventory so monthly imports compound
    setItems((prev) => [...prev, ...newItems]);
    toast(`Imported ${newItems.length} rows. Model updated.`);
  }

  function onGTMAction(strategyTitle: string, intent: string) {
    toast(`${strategyTitle}: ${intent}`);
  }

  return (
    <div className="app">
      <TopNav page={page} setPage={setPage} />

      <div className="container">
        <UploadPanel onLoaded={onLoaded} />
        <KPIBar items={items} selectedCount={selectedItems.length} />

        <div id="export-area">
          {page === "dashboard" && (
            <>
              <FlowDiagram />
              <FiltersPanel filters={filters} setFilters={setFilters} />
              <Top50Table
                items={items}
                filters={filters}
                selected={selectedIds}
                toggleSelect={toggleSelect}
                onOpen={(x) => setDrawerItem(x)}
              />
            </>
          )}

          {page === "top50" && (
            <>
              <FiltersPanel filters={filters} setFilters={setFilters} />
              <Top50Table
                items={items}
                filters={filters}
                selected={selectedIds}
                toggleSelect={toggleSelect}
                onOpen={(x) => setDrawerItem(x)}
              />
            </>
          )}

          {page === "gtm" && (
            <GoToMarket selectedItems={selectedItems} scenario={scenario} onAction={onGTMAction} />
          )}

          {page === "sim" && (
            <ROISimulator scenario={scenario} setScenario={setScenario} />
          )}

          {page === "export" && (
            <ExportButtons toast={toast} />
          )}

          {page === "demo" && (
            <DemoNarration />
          )}
        </div>
      </div>

      <ItemDrawer
        item={drawerItem}
        onClose={() => setDrawerItem(undefined)}
        onSelectToggle={toggleSelect}
        selected={drawerItem ? selectedIds.has(drawerItem.id) : false}
      />

      {toastMsg && <Toast message={toastMsg} onClose={() => setToastMsg("")} />}
    </div>
  );
}
