  <script>
    const chat = document.getElementById("chat");
    const input = document.getElementById("input");
    const send = document.getElementById("send");
    const provider = document.getElementById("provider");
    const model = document.getElementById("model");
    const truthMode = document.getElementById("truthMode");

    const modelsByProvider = {
      openai: ["gpt-5.2-thinking", "gpt-5.2-instant", "gpt-5.2-pro"],
      claude: ["claude-opus-4-1", "claude-sonnet-4-1"],
      gemini: ["gemini-2.5-flash", "gemini-2.5-pro"]
    };

    const state = {
      messages: [
        { role: "system", content: "You are SmartBrain. Be precise. If truthMode is on, rely on citations and admit uncertainty." }
      ]
    };

    function setModelOptions(){
      model.innerHTML = "";
      for(const m of modelsByProvider[provider.value]){
        const opt = document.createElement("option");
        opt.value = m; opt.textContent = m;
        model.appendChild(opt);
      }
      if(provider.value !== "openai") truthMode.checked = false;
      truthMode.disabled = provider.value !== "openai";
    }

    function addMsg(role, text, citations){
      const box = document.createElement("div");
      box.className = "msg " + (role === "user" ? "user" : "asst");
      box.textContent = text;

      if(citations && citations.length){
        const c = document.createElement("div");
        c.className = "citations";
        c.innerHTML = "<div class='meta'>Sources</div>" + citations.map(x =>
          `<div>• <a href="${x.url}" target="_blank" rel="noreferrer">${x.title}</a></div>`
        ).join("");
        box.appendChild(c);
      }

      chat.appendChild(box);
      chat.scrollTop = chat.scrollHeight;
    }

    async function go(){
      const text = input.value.trim();
      if(!text) return;

      input.value = "";
      addMsg("user", text);

      state.messages.push({ role: "user", content: text });
      send.disabled = true;

      try{
        const resp = await fetch("/api/chat", {
          method: "POST",
          headers: {"Content-Type":"application/json"},
          body: JSON.stringify({
            provider: provider.value,
            model: model.value,
            messages: state.messages,
            truthMode: truthMode.checked
          })
        });

        const data = await resp.json();
        if(!resp.ok) throw new Error(data.error || "Request failed");

        addMsg("assistant", data.text || "(empty)", data.citations || []);
        state.messages.push({ role: "assistant", content: data.text || "" });
      } catch(e){
        addMsg("assistant", "Error: " + e.message);
      } finally{
        send.disabled = false;
      }
    }

    send.onclick = go;
    input.addEventListener("keydown", (e) => {
      if(e.key === "Enter" && !e.shiftKey){ e.preventDefault(); go(); }
    });

    provider.onchange = setModelOptions;
    setModelOptions();
  </script>
