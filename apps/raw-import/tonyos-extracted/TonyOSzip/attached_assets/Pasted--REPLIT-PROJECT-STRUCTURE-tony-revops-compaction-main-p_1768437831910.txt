3️⃣ rules.py (ALL BUSINESS LOGIC LIVES HERE)
MANUFACTURING_REQUIRED = [
    "manufacturing","production","processing","pressing",
    "forming","compaction","powder","tablet","pellet",
    "sinter","scale-up","formulation"
]

EXCLUDE_ALWAYS = [
    "clinic","practice","dealer","dealership",
    "distributor","reseller","retail",
    "service","repair","maintenance",
    "equipment sales","pressure washer",
    "animal hospital","spring manufacturer"
]

ENGINEERING_SIGNALS = [
    "tablet press","tooling","die","punch","compression",
    "powder metallurgy","ceramic","pellet","pressing"
]

SCIENTIFIC_SIGNALS = [
    "r&d","research","formulation","analytical",
    "pilot","scale-up","material science","lab"
]

AUTOMOTIVE_REQUIRED = [
    "airbag","inflator","gas generant",
    "energetic","propellant","pyrotechnic"
]

def contains_any(text, keywords):
    return any(k in text for k in keywords)

def classify_fit(text):
    eng = contains_any(text, ENGINEERING_SIGNALS)
    sci = contains_any(text, SCIENTIFIC_SIGNALS)

    if eng and sci:
        return "Both"
    if eng:
        return "Natoli Engineering"
    if sci:
        return "Natoli Scientific"
    return None

def assign_tier(text):
    if "pharma" in text or "tablet" in text:
        return "Tier 1"
    if "ceramic" in text or "powder metallurgy" in text:
        return "Tier 2"
    if "materials" in text or "mining" in text:
        return "Tier 3"
    return "Tier 4"

def fit_reason(fit, tier):
    if fit == "Natoli Engineering":
        return "Production-scale compaction, tooling, or pressing requirements"
    if fit == "Natoli Scientific":
        return "R&D, formulation, or scale-up activities aligned with Natoli Scientific"
    if fit == "Both":
        return "R&D to production workflow requiring Scientific and Engineering support"
    return ""

4️⃣ processor.py (THE ENGINE)
import pandas as pd
import requests
from bs4 import BeautifulSoup
from rules import *
from territories import assign_territory
import time

HEADERS = {"User-Agent":"Mozilla/5.0"}
TIMEOUT = 8

def get_site_text(url):
    try:
        r = requests.get(url, headers=HEADERS, timeout=TIMEOUT)
        soup = BeautifulSoup(r.text,"html.parser")
        return soup.get_text(" ").lower()
    except:
        return ""

def process_csv(input_csv):
    df = pd.read_csv(input_csv)

    good = []
    bad = []

    for _, row in df.iterrows():
        website = str(row.get("Website","")).strip()
        state = str(row.get("Company State","")).title()

        if not website.startswith("http"):
            continue

        text = get_site_text(website)

        if contains_any(text, EXCLUDE_ALWAYS):
            continue

        if not contains_any(text, MANUFACTURING_REQUIRED):
            continue

        if row.get("Industry","").lower() == "automotive":
            if not contains_any(text, AUTOMOTIVE_REQUIRED):
                continue

        fit = classify_fit(text)
        if not fit:
            continue

        tier = assign_tier(text)
        territory = assign_territory(state)

        row["Natoli_Fit"] = fit
        row["Tier"] = tier
        row["Territory"] = territory
        row["Fit_Reason"] = fit_reason(fit, tier)

        good.append(row)
        time.sleep(0.5)

    clean_df = pd.DataFrame(good)
    clean_df.to_csv("TONY_REVOPS_CLEANED.csv", index=False)

    return clean_df
