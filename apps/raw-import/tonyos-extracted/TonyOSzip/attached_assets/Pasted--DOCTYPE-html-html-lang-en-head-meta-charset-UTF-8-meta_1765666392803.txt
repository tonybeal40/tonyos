<script>
  // ========= CONFIG =========
  const API_URL = "PASTE_YOUR_WEB_APP_URL_HERE";
  const API_TOKEN = "PASTE_YOUR_TOKEN_HERE";

  // Local settings
  const SETTINGS_KEY = "tonyos_money_settings_v1";
  const defaultSettings = {
    weeklyBudget: 250,
    bufferCash: 0,
    mode: "Normal"
  };

  function money(n){
    const v = Number(n || 0);
    return v.toLocaleString(undefined, { style:"currency", currency:"USD" });
  }

  function loadSettings(){
    try { return { ...defaultSettings, ...(JSON.parse(localStorage.getItem(SETTINGS_KEY) || "{}")) }; }
    catch { return { ...defaultSettings }; }
  }
  function saveSettings(s){
    localStorage.setItem(SETTINGS_KEY, JSON.stringify(s));
  }

  // ======== Date helpers ========
  function startOfWeek(d){
    const dt = new Date(d);
    const day = dt.getDay(); // 0 Sun
    const diff = (day + 6) % 7; // make Monday start
    dt.setDate(dt.getDate() - diff);
    dt.setHours(0,0,0,0);
    return dt;
  }
  function endOfWeek(d){
    const s = startOfWeek(d);
    const e = new Date(s);
    e.setDate(e.getDate() + 7);
    return e;
  }

  function daysBetween(a, b){
    const ms = 24*60*60*1000;
    return Math.ceil((b - a) / ms);
  }

  async function apiGet(action, params = {}){
    const url = new URL(API_URL);
    url.searchParams.set("action", action);
    url.searchParams.set("token", API_TOKEN);
    Object.entries(params).forEach(([k,v]) => url.searchParams.set(k, v));
    const res = await fetch(url.toString());
    return await res.json();
  }

  async function apiPost(payload){
    const res = await fetch(API_URL, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ ...payload, token: API_TOKEN })
    });
    return await res.json();
  }

  function setBar(el, pct){
    const p = Math.max(0, Math.min(100, pct));
    el.style.width = p + "%";
    if (p < 70) el.style.background = "rgba(34,197,94,.9)";
    else if (p < 95) el.style.background = "rgba(251,191,36,.95)";
    else el.style.background = "rgba(225,29,72,.95)";
  }

  function renderRows(rows){
    const tb = document.getElementById("rows");
    tb.innerHTML = "";
    rows.slice(0, 40).forEach(r => {
      const tr = document.createElement("tr");
      const dt = new Date(r.createdAt);
      const amount = Number(r.amount || 0) * (r.type === "expense" ? -1 : 1);
      tr.innerHTML = `
        <td>${dt.toLocaleString()}</td>
        <td>${r.type}</td>
        <td>${r.category || ""}</td>
        <td>${r.merchant || ""}</td>
        <td>${money(amount)}</td>
        <td>${r.note || ""}</td>
      `;
      tb.appendChild(tr);
    });
  }

  function computeKPIs(rows, settings){
    const now = new Date();
    const wkStart = startOfWeek(now);
    const wkEnd = endOfWeek(now);
    const daysLeft = Math.max(1, daysBetween(now, wkEnd));

    let spentWeek = 0;
    let income14 = 0;
    let expense14 = 0;

    const d14 = new Date(now);
    d14.setDate(d14.getDate() - 14);

    rows.forEach(r => {
      const dt = new Date(r.createdAt);
      const amt = Number(r.amount || 0);

      // Week spend
      if (dt >= wkStart && dt < wkEnd && r.type === "expense") {
        spentWeek += amt;
      }

      // 14 day net
      if (dt >= d14) {
        if (r.type === "income") income14 += amt;
        if (r.type === "expense") expense14 += amt;
      }
    });

    const weeklyBudget = Number(settings.weeklyBudget || 0);
    const remainingWeek = weeklyBudget - spentWeek;
    const dailyAllowance = remainingWeek / daysLeft;

    const runway14 = income14 - expense14; // net for last 14
    const pacePct = weeklyBudget > 0 ? (spentWeek / weeklyBudget) * 100 : 0;

    return { weeklyBudget, spentWeek, remainingWeek, dailyAllowance, daysLeft, runway14, pacePct };
  }

  async function refresh(){
    const settings = loadSettings();

    // Quick interactive prompt to set weekly budget if not set
    if (!settings.weeklyBudget || settings.weeklyBudget <= 0) {
      const v = prompt("Set weekly budget (number)", "250");
      settings.weeklyBudget = Number(v || 250);
      saveSettings(settings);
    }

    document.getElementById("weeklyBudgetVal").textContent = money(settings.weeklyBudget);
    document.getElementById("modeLabel").textContent = settings.mode || "Normal";

    const ping = await apiGet("ping");
    if (!ping.ok) {
      document.getElementById("paceMsg").textContent = "API error: " + (ping.error || "unknown");
      document.getElementById("paceMsg").className = "small bad";
      return;
    }

    const data = await apiGet("list", { limit: 800 });
    if (!data.ok) {
      document.getElementById("paceMsg").textContent = "List error: " + (data.error || "unknown");
      document.getElementById("paceMsg").className = "small bad";
      return;
    }

    const rows = data.rows || [];
    const k = computeKPIs(rows, settings);

    document.getElementById("spentWeekVal").textContent = money(k.spentWeek);
    document.getElementById("dailyAllowanceVal").textContent = money(k.dailyAllowance);
    document.getElementById("runwayVal").textContent = money(k.runway14);

    document.getElementById("daysLeftMsg").textContent = k.daysLeft + " day(s) left in the week";
    setBar(document.getElementById("spentBar"), k.pacePct);

    const pace = document.getElementById("paceMsg");
    if (k.pacePct <= 70){
      pace.textContent = "On pace. Keep it boring and you win";
      pace.className = "small ok";
      settings.mode = "Normal";
    } else if (k.pacePct < 95){
      pace.textContent = "Close to the line. Tighten up now";
      pace.className = "small warn";
      settings.mode = "Tight";
    } else {
      pace.textContent = "Over budget pace. Bare minimum mode until next week";
      pace.className = "small bad";
      settings.mode = "Bare Minimum";
    }
    saveSettings(settings);
    document.getElementById("modeLabel").textContent = settings.mode;

    renderRows(rows);
  }

  document.getElementById("refreshBtn").addEventListener("click", refresh);

  document.getElementById("addBtn").addEventListener("click", async () => {
    const type = document.getElementById("type").value;
    const amount = Number(document.getElementById("amount").value || 0);
    const category = document.getElementById("category").value.trim();
    const merchant = document.getElementById("merchant").value.trim();
    const note = document.getElementById("note").value.trim();

    const status = document.getElementById("addStatus");
    status.textContent = "";

    if (!amount || amount <= 0) {
      status.textContent = "Amount must be > 0";
      status.className = "small bad";
      return;
    }
    if (!category) {
      status.textContent = "Category required";
      status.className = "small bad";
      return;
    }

    const res = await apiPost({
      createdAt: new Date().toISOString(),
      amount,
      category,
      merchant,
      note,
      type
    });

    if (!res.ok) {
      status.textContent = "Add failed: " + (res.error || "unknown");
      status.className = "small bad";
      return;
    }

    document.getElementById("amount").value = "";
    document.getElementById("category").value = "";
    document.getElementById("merchant").value = "";
    document.getElementById("note").value = "";

    status.textContent = "Saved";
    status.className = "small ok";
    refresh();
  });

  refresh();
</script>
