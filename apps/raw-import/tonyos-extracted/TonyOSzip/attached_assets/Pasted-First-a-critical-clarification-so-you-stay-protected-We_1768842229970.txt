import requests
import pandas as pd
import re
import time
from bs4 import BeautifulSoup
from urllib.parse import urlparse
from openai import OpenAI

# ===============================
# CONFIG
# ===============================

BING_API_KEY = "YOUR_BING_API_KEY"
OPENAI_API_KEY = "YOUR_OPENAI_API_KEY"

USER_AGENT = "NatoliCompanyResearch/1.0 (+public-data)"
REQUEST_DELAY = 2.0

client = OpenAI(api_key=OPENAI_API_KEY)

INPUT_FILE = "natoli_net_new_targets.csv"  # or your source file
OUTPUT_FILE = "natoli_company_profiles.csv"

BING_ENDPOINT = "https://api.bing.microsoft.com/v7.0/search"
HEADERS = {
    "Ocp-Apim-Subscription-Key": BING_API_KEY,
    "User-Agent": USER_AGENT
}

# ===============================
# HELPERS
# ===============================

def bing_search(query):
    params = {"q": query, "count": 5}
    r = requests.get(BING_ENDPOINT, headers=HEADERS, params=params)
    r.raise_for_status()
    return r.json().get("webPages", {}).get("value", [])

def extract_phone(text):
    if not text:
        return ""
    match = re.search(r"(\(?\d{3}\)?[-.\s]\d{3}[-.\s]\d{4})", text)
    return match.group(1) if match else ""

def extract_address(text):
    if not text:
        return "", "", ""
    city_state = re.search(r"([A-Z][a-zA-Z\s]+),\s([A-Z]{2})", text)
    if city_state:
        return city_state.group(1), city_state.group(2), ""
    return "", "", ""

def fetch_site_text(url):
    try:
        r = requests.get(url, headers={"User-Agent": USER_AGENT}, timeout=6)
        if r.status_code != 200:
            return ""
        soup = BeautifulSoup(r.text, "html.parser")
        return soup.get_text(" ", strip=True)
    except:
        return ""

def find_linkedin(results):
    for r in results:
        url = r.get("url", "")
        if "linkedin.com/company" in url:
            return url
    return ""

# ===============================
# OPENAI CLEANUP / CONFIDENCE
# ===============================

def assess_profile_confidence(company, website, phone, city, state, linkedin):
    prompt = f"""
You are validating a public company profile.

Company: {company}
Website: {website}
Phone: {phone}
City: {city}
State: {state}
LinkedIn: {linkedin}

Return one word only:
High, Medium, or Low

High = most fields present and consistent
Medium = partial but reasonable
Low = sparse or uncertain
"""
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        temperature=0
    )
    return response.choices[0].message.content.strip()

# ===============================
# PIPELINE
# ===============================

df = pd.read_csv(INPUT_FILE)
rows = []

for _, row in df.iterrows():
    company = row["Company"]

    results = bing_search(company)
    website = results[0]["url"] if results else ""
    snippet = results[0]["snippet"] if results else ""

    site_text = fetch_site_text(website)
    combined_text = snippet + " " + site_text

    phone = extract_phone(combined_text)
    city, state, address = extract_address(combined_text)
    linkedin = find_linkedin(results)

    confidence = assess_profile_confidence(
        company, website, phone, city, state, linkedin
    )

    rows.append({
        "Company": company,
        "Website": website,
        "Phone": phone,
        "City": city,
        "State": state,
        "Address": address,
        "LinkedIn": linkedin,
        "Source": urlparse(website).netloc,
        "Profile_Confidence": confidence
    })

    time.sleep(REQUEST_DELAY)

out_df = pd.DataFrame(rows)
out_df.to_csv(OUTPUT_FILE, index=False)

print("Company profile enrichment complete:", OUTPUT_FILE)
