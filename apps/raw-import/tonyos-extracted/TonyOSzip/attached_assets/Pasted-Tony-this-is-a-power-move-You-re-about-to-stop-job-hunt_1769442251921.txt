STEP 4 – main.py (Job Finder Engine)
import requests
from bs4 import BeautifulSoup
import pandas as pd
from datetime import datetime

KEYWORDS = [
    "revenue operations",
    "revops",
    "business development director",
    "head of growth",
    "growth operations",
    "sales operations",
    "gtm",
    "crm",
    "manufacturing",
    "b2b"
]

LOCATION = "remote OR hybrid"
MIN_FIT_SCORE = 3

HEADERS = {
    "User-Agent": "Mozilla/5.0"
}

def score_job(text):
    return sum(1 for k in KEYWORDS if k.lower() in text.lower())

def search_google(keyword):
    query = f"{keyword} jobs {LOCATION}"
    url = f"https://www.google.com/search?q={query.replace(' ', '+')}"
    html = requests.get(url, headers=HEADERS).text
    soup = BeautifulSoup(html, "html.parser")

    results = []
    for block in soup.select("div.BVG0Nb"):
        title = block.text.strip()
        fit = score_job(title)
        if fit >= MIN_FIT_SCORE:
            results.append({
                "Keyword": keyword,
                "Title": title,
                "Fit Score": fit
            })
    return results

def main():
    all_jobs = []
    for k in KEYWORDS:
        print(f"Searching: {k}")
        jobs = search_google(k)
        all_jobs.extend(jobs)

    df = pd.DataFrame(all_jobs)
    df = df.drop_duplicates(subset=["Title"])
    df = df.sort_values("Fit Score", ascending=False)

    filename = f"tony_job_matches_{datetime.now().strftime('%Y%m%d')}.csv"
    df.to_csv(filename, index=False)

    print("\n==============================")
    print("TOP MATCHES")
    print("==============================")
    print(df.head(15))
    print(f"\nSaved file: {filename}")

if __name__ == "__main__":
    main()
