import requests
import pandas as pd
import time
from bs4 import BeautifulSoup
from urllib.parse import urlparse

# ===============================
# CONFIG
# ===============================

BING_API_KEY = "YOUR_BING_KEY"
GOOGLE_API_KEY = "YOUR_GOOGLE_KEY"
GOOGLE_CSE_ID = "YOUR_CSE_ID"

HEADERS = {
    "Ocp-Apim-Subscription-Key": BING_API_KEY
}

MAX_RESULTS_PER_QUERY = 10
REQUEST_DELAY = 1.5

# ===============================
# NATOLI FIT CLASSIFIER
# ===============================

def natoli_fit_classifier(text):
    if not text:
        return "No Fit", "No public manufacturing signal"

    t = text.lower()

    disqualifiers = [
        "injectable only","infusion","admixture",
        "clinic","wellness","pellet","trt"
    ]
    if any(d in t for d in disqualifiers):
        return "No Fit", "Sterile or non-compression focused operation"

    engineering = any(k in t for k in [
        "tablet","capsule","solid oral","manufacturing",
        "production","press","throughput","batch"
    ])

    scientific = any(k in t for k in [
        "formulation","ms&t","characterization",
        "scale-up","analytical","powder"
    ])

    if engineering and scientific:
        return "Both", "Production compression with formulation or scale-up activity"
    if engineering:
        return "Engineering", "Tablet production where press uptime and tooling matter"
    if scientific:
        return "Scientific", "Compaction, formulation, or material science focus"

    return "No Fit", "No tablet compression or compaction relevance"

# ===============================
# SEARCH FUNCTIONS
# ===============================

def bing_search(query):
    url = "https://api.bing.microsoft.com/v7.0/search"
    params = {"q": query, "count": MAX_RESULTS_PER_QUERY}
    r = requests.get(url, headers=HEADERS, params=params)
    r.raise_for_status()
    return r.json().get("webPages", {}).get("value", [])

def google_search(query):
    url = "https://www.googleapis.com/customsearch/v1"
    params = {
        "key": GOOGLE_API_KEY,
        "cx": GOOGLE_CSE_ID,
        "q": query
    }
    r = requests.get(url, params=params)
    r.raise_for_status()
    return r.json().get("items", [])

# ===============================
# WEBSITE TEXT FETCH (ROBOTS SAFE)
# ===============================

def fetch_website_text(url):
    try:
        r = requests.get(url, timeout=5, headers={"User-Agent": "Mozilla/5.0"})
        soup = BeautifulSoup(r.text, "html.parser")
        return soup.get_text(" ", strip=True)[:8000]
    except:
        return ""

# ===============================
# MAIN PIPELINE
# ===============================

def run_pipeline(industry_queries):
    rows = []

    for industry in industry_queries:
        results = bing_search(industry)

        for r in results:
            company = r.get("name")
            website = r.get("url")
            snippet = r.get("snippet", "")
            domain = urlparse(website).netloc

            site_text = fetch_website_text(website)
            combined_text = snippet + " " + site_text

            fit, reason = natoli_fit_classifier(combined_text)

            rows.append({
                "Company": company,
                "City": "",
                "State": "",
                "Phone": "",
                "Contact": "",
                "Website": website,
                "LinkedIn": "",
                "Natoli Fit": fit,
                "Reason": reason,
                "Industry": industry,
                "Source": domain
            })

            time.sleep(REQUEST_DELAY)

    return pd.DataFrame(rows)

# ===============================
# RUN
# ===============================

ALL_INDUSTRIES = ENGINEERING_INDUSTRIES + SCIENTIFIC_INDUSTRIES

df = run_pipeline(ALL_INDUSTRIES)

df.drop_duplicates(subset=["Company","Website"], inplace=True)
df.to_csv("natoli_us_targets.csv", index=False)

print("CSV created: natoli_us_targets.csv")
