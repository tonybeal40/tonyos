        r = requests.get(url, headers=HEADERS, timeout=TIMEOUT)

        text = soup.get_text(separator=" ").lower()
        return text

        return ""

    return any(k in text for k in keywords)

def scan_site(base_url):
    full_text = ""
    for page in SUBPAGES:
        full_text += " " + fetch_text(base_url + page)
        time.sleep(0.3)
    return full_text

def contains_any(text, terms):
    return any(t in text for t in terms)

def proximity_match(text, materials, verbs, window=40):
    for m in materials:
        for v in verbs:
            pattern = rf"{m}.{{0,{window}}}{v}|{v}.{{0,{window}}}{m}"
            if re.search(pattern, text):
                return True
    return False

        row["Reason"] = "No verified compaction manufacturing"

    text = ""
    for p in SUBPAGES:
        text += " " + fetch_text(base + p)
        time.sleep(0.3)
    return text

def contains_any(text, terms):
    return any(t in text for t in terms)

def proximity_match(text, materials, verbs, window=40):
    for m in materials:
        for v in verbs:
            if re.search(rf"{m}.{{0,{window}}}{v}|{v}.{{0,{window}}}{m}", text):
                return True
    return False

def assign_territory(state):
    s = str(state).lower()
    for territory, states in TERRITORIES.items():
        if s in states:
            return territory
    return "Unassigned"

def natoli_route(text):
    eng = contains_any(text, ENGINEERING_TERMS)
    sci = contains_any(text, SCIENTIFIC_TERMS)
    if eng and sci:
        return "Both"
    if eng:
        return "Natoli Engineering"
    if sci:
        return "Natoli Scientific"
    return "Undetermined"

# ================= MAIN =================

df = pd.read_csv(INPUT_FILE)

keep, review, exclude = [], [], []

for _, row in df.iterrows():
    website = str(row.get("Website", "")).strip()
    industry = str(row.get("Industry", "")).lower()
    state = row.get("Company State", "")

    row["Territory"] = assign_territory(state)

    if not website.startswith("http"):
        row["Decision"] = "EXCLUDE"
        row["Confidence"] = "Low"
        row["Explanation"] = "No website available"
        exclude.append(row)
        continue

    text = scan_site(website)

    if contains_any(text, BLACKLIST):
        row["Decision"] = "EXCLUDE"
        row["Confidence"] = "Low"
        row["Explanation"] = "Service, dealer, or non-manufacturing business"
        exclude.append(row)
        continue

    manuf = contains_any(text, MANUFACTURING_TERMS)
    process = proximity_match(text, MATERIAL_TERMS, PROCESS_VERBS)
    industry_match = contains_any(industry, TIER1_INDUSTRIES)

    score = 0
    if manuf: score += 30
    if process: score += 40
    if industry_match: score += 30

    row["Score"] = score
    row["Natoli_Route"] = natoli_route(text)

    if score >= 70:
        row["Decision"] = "KEEP"
        row["Confidence"] = "High"
        row["Explanation"] = "Manufacturing ownership and material transformation confirmed"
        keep.append(row)
    elif score >= 40:
        row["Decision"] = "REVIEW"
        row["Confidence"] = "Medium"
        row["Explanation"] = "Partial manufacturing or industry alignment"
        review.append(row)
    else:
        row["Decision"] = "EXCLUDE"
        row["Confidence"] = "Low"
        row["Explanation"] = "No verified compaction manufacturing"
        exclude.append(row)

    time.sleep(SLEEP)
