🚀 main.py (core SaaS engine)

Copy this exactly:

from flask import Flask, render_template_string, request, send_file, redirect
from openpyxl import load_workbook
import pandas as pd
import os
import uuid

app = Flask(__name__)

TEMPLATE_DIR = "templates"
EXPORT_DIR = "exports"

os.makedirs(EXPORT_DIR, exist_ok=True)

# ---------- HTML UI ----------
HTML = """
<!DOCTYPE html>
<html>
<head>
<title>Excel SaaS Generator</title>
<style>
body {font-family: Arial; background:#111; color:white; padding:40px;}
.card {background:#222;padding:20px;border-radius:10px;}
button {padding:10px 20px;margin-top:10px;background:#00b894;color:white;border:none;}
select,input {padding:8px;width:100%;}
</style>
</head>
<body>

<h2>Excel Report Engine</h2>

<div class="card">

<form action="/generate" method="post">

<label>Select Template:</label>
<select name="template">
{% for t in templates %}
<option value="{{t}}">{{t}}</option>
{% endfor %}
</select>

<label>Sample Data Value:</label>
<input name="value" placeholder="Enter value">

<button type="submit">Generate Report</button>

</form>

</div>

<br>

<div class="card">

<h3>Upload Template</h3>

<form action="/upload" method="post" enctype="multipart/form-data">
<input type="file" name="file">
<button type="submit">Upload Template</button>
</form>

</div>

</body>
</html>
"""

# ---------- Homepage ----------
@app.route("/")
def home():
    templates = os.listdir(TEMPLATE_DIR)
    return render_template_string(HTML, templates=templates)

# ---------- Upload Template ----------
@app.route("/upload", methods=["POST"])
def upload():
    file = request.files["file"]
    if file:
        file.save(os.path.join(TEMPLATE_DIR, file.filename))
    return redirect("/")

# ---------- Excel Generator ----------
@app.route("/generate", methods=["POST"])
def generate():

    template_name = request.form["template"]
    value = request.form.get("value", "Demo Data")

    template_path = os.path.join(TEMPLATE_DIR, template_name)

    wb = load_workbook(template_path)
    ws = wb.active

    # Example injection cell
    ws["B5"] = value

    export_name = f"{uuid.uuid4()}.xlsx"
    export_path = os.path.join(EXPORT_DIR, export_name)

    wb.save(export_path)

    return send_file(export_path, as_attachment=True)

# ---------- Run ----------
app.run(host="0.0.0.0", port=3000)
