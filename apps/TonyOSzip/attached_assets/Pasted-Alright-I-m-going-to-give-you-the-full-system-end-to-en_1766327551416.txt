🔌 STEP 2: APOLLO CONNECTION
/server/apollo.js
import fetch from "node-fetch";

const APOLLO_API_KEY = process.env.APOLLO_API_KEY;

export async function enrichCompany(domain) {
  const res = await fetch("https://api.apollo.io/v1/companies/search", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "X-Api-Key": APOLLO_API_KEY
    },
    body: JSON.stringify({
      q: domain,
      page: 1,
      per_page: 1
    })
  });

  const data = await res.json();
  return data.companies?.[0] || null;
}

export async function getPersonas(companyId, titles) {
  const res = await fetch("https://api.apollo.io/v1/people/search", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "X-Api-Key": APOLLO_API_KEY
    },
    body: JSON.stringify({
      organization_ids: [companyId],
      person_titles: titles,
      page: 1,
      per_page: 25
    })
  });

  const data = await res.json();
  return data.people || [];
}

⚖️ STEP 5: PERSONA SCORING ENGINE
/server/scoring.js
import { TOP_PERSONAS } from "./personas.js";

export function scorePersonas(people) {
  let score = 0;
  let matched = [];

  people.forEach(p => {
    const title = p.title;
    if (TOP_PERSONAS[title]) {
      score += TOP_PERSONAS[title];
      matched.push(title);
    }
  });

  return { score, matched };
}

export function assignTier(score) {
  if (score >= 10) return "A";
  if (score >= 6) return "B";
  return "C";
}

🔀 STEP 6: NATOLI ROUTING LOGIC
/server/routing.js
import { INDUSTRIES } from "./industries.js";

export function routeNatoli(naicsCodes) {
  let eng = false;
  let sci = false;

  Object.values(INDUSTRIES).forEach(i => {
    if (i.naics.some(code => naicsCodes?.includes(code))) {
      if (i.division.includes("engineering")) eng = true;
      if (i.division.includes("scientific")) sci = true;
    }
  });

  if (eng && sci) return "BOTH";
  if (eng) return "Natoli Engineering";
  if (sci) return "Natoli Scientific";
  return "UNQUALIFIED";
}

🎯 STEP 7: PAIN POINT ENGINE
/server/pains.js
export function painPoints(industry, division) {
  const base = {
    scientific: [
      "Scale-up risk",
      "Analytical data gaps",
      "Method transfer uncertainty"
    ],
    engineering: [
      "Tooling wear",
      "Downtime",
      "Inconsistent compression"
    ]
  };

  if (division === "BOTH") {
    return [...base.scientific, ...base.engineering];
  }

  return base[division.toLowerCase()] || [];
}

🚀 STEP 9: MAIN SERVER
/server/index.js
import express from "express";
import { enrichCompany, getPersonas } from "./apollo.js";
import { scorePersonas, assignTier } from "./scoring.js";
import { routeNatoli } from "./routing.js";
import { painPoints } from "./pains.js";
import { getCached, setCache } from "./cache.js";
import { TOP_PERSONAS } from "./personas.js";

const app = express();
app.use(express.json());

app.post("/analyze", async (req, res) => {
  const { domain } = req.body;

  if (getCached(domain)) {
    return res.json(getCached(domain));
  }

  const company = await enrichCompany(domain);
  if (!company) return res.status(404).json({ error: "Company not found" });

  const people = await getPersonas(
    company.id,
    Object.keys(TOP_PERSONAS)
  );

  const personaData = scorePersonas(people);
  const tier = assignTier(personaData.score);
  const division = routeNatoli(company.naics_codes || []);
  const pains = painPoints(company.industry, division);

  const output = {
    company: company.name,
    website: company.website_url,
    linkedin: company.linkedin_url,
    employees: company.estimated_num_employees,
    revenue: company.annual_revenue,
    naics: company.naics_codes,
    sic: company.sic_codes,
    industry: company.industry,
    division_fit: division,
    persona_score: personaData.score,
    matched_personas: personaData.matched,
    tier,
    pain_points: pains,
    why_natoli: division === "Natoli Scientific"
      ? "Analytical rigor and scale-up confidence"
      : division === "Natoli Engineering"
      ? "Precision tooling and production reliability"
      : "End-to-end compaction support"
  };

  setCache(domain, output);
  res.json(output);
});

app.listen(3000, () => console.log("Tony CRM running"));
