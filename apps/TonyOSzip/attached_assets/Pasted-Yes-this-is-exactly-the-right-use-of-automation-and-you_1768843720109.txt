import requests
import pandas as pd
import re
import time
from bs4 import BeautifulSoup
from urllib.parse import urlparse
from openai import OpenAI

# ===============================
# CONFIG
# ===============================

BING_API_KEY = "YOUR_BING_API_KEY"
OPENAI_API_KEY = "YOUR_OPENAI_KEY"

USER_AGENT = "NatoliAccountRecovery/1.0 (+public-research)"
REQUEST_DELAY = 2.0

INPUT_FILE = "/mnt/data/natoli_targets.csv"
OUTPUT_FILE = "natoli_lapsed_account_enrichment.csv"

BING_ENDPOINT = "https://api.bing.microsoft.com/v7.0/search"
HEADERS = {
    "Ocp-Apim-Subscription-Key": BING_API_KEY,
    "User-Agent": USER_AGENT
}

client = OpenAI(api_key=OPENAI_API_KEY)

# ===============================
# SEARCH
# ===============================

def bing_search(query):
    params = {"q": query, "count": 5}
    r = requests.get(BING_ENDPOINT, headers=HEADERS, params=params)
    r.raise_for_status()
    return r.json().get("webPages", {}).get("value", [])

# ===============================
# EXTRACTION HELPERS
# ===============================

def extract_phone(text):
    if not text:
        return ""
    match = re.search(r"(\(?\d{3}\)?[-.\s]\d{3}[-.\s]\d{4})", text)
    return match.group(1) if match else ""

def extract_city_state(text):
    if not text:
        return "", ""
    match = re.search(r"([A-Z][a-zA-Z\s]+),\s([A-Z]{2})", text)
    return (match.group(1), match.group(2)) if match else ("", "")

def fetch_site_text(url):
    try:
        r = requests.get(url, headers={"User-Agent": USER_AGENT}, timeout=6)
        if r.status_code != 200:
            return ""
        soup = BeautifulSoup(r.text, "html.parser")
        return soup.get_text(" ", strip=True)
    except:
        return ""

def find_linkedin_company(results):
    for r in results:
        url = r.get("url", "")
        if "linkedin.com/company" in url:
            return url
    return ""

# ===============================
# CONFIDENCE ASSESSMENT (AI)
# ===============================

def profile_confidence(company, website, phone, city, state, linkedin):
    prompt = f"""
You are validating a public company profile.

Company: {company}
Website: {website}
Phone: {phone}
City: {city}
State: {state}
LinkedIn: {linkedin}

Return exactly one word:
High, Medium, or Low
"""
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        temperature=0
    )
    return response.choices[0].message.content.strip()

# ===============================
# PIPELINE
# ===============================

df = pd.read_csv(INPUT_FILE)
rows = []

for _, row in df.iterrows():
    company = row["Company"]

    results = bing_search(company)
    if not results:
        continue

    website = results[0].get("url", "")
    snippet = results[0].get("snippet", "")
    site_text = fetch_site_text(website)

    combined_text = snippet + " " + site_text

    phone = extract_phone(combined_text)
    city, state = extract_city_state(combined_text)
    linkedin = find_linkedin_company(results)

    confidence = profile_confidence(
        company, website, phone, city, state, linkedin
    )

    rows.append({
        "Company": company,
        "Website": website,
        "Phone": phone,
        "City": city,
        "State": state,
        "LinkedIn": linkedin,
        "Profile Confidence": confidence
    })

    time.sleep(REQUEST_DELAY)

out_df = pd.DataFrame(rows)
out_df.to_csv(OUTPUT_FILE, index=False)

print("Lapsed account enrichment complete:", OUTPUT_FILE)
