1️⃣ Add Apollo + Scoring into the CRM save flow
Create new file
/server/enrich-and-save.js
import { enrichCompany, getPersonas } from "./apollo.js";
import { scorePersonas, assignTier } from "./scoring.js";
import { routeNatoli } from "./routing.js";
import { painPoints } from "./pains.js";
import { TOP_PERSONAS } from "./personas.js";
import { createAccount } from "./accounts.js";

export async function enrichAndSave(domain) {
  const company = await enrichCompany(domain);
  if (!company) throw new Error("Company not found in Apollo");

  const people = await getPersonas(
    company.id,
    Object.keys(TOP_PERSONAS)
  );

  const personaData = scorePersonas(people);
  const tier = assignTier(personaData.score);
  const division = routeNatoli(company.naics_codes || []);
  const pains = painPoints(company.industry, division);

  const accountData = {
    company_name: company.name,
    domain: domain,
    website: company.website_url,
    linkedin_url: company.linkedin_url,
    industry: company.industry,
    naics: (company.naics_codes || []).join(","),
    sic: (company.sic_codes || []).join(","),
    employees: company.estimated_num_employees,
    revenue: company.annual_revenue,
    division_fit: division,
    persona_score: personaData.score,
    tier,
    pain_points: pains.join(", "),
    why_natoli:
      division === "Natoli Scientific"
        ? "Analytical rigor and scale-up confidence"
        : division === "Natoli Engineering"
        ? "Precision tooling and production reliability"
        : "End-to-end compaction support",
    status: "New"
  };

  const result = createAccount(accountData);
  return { id: result.lastInsertRowid, ...accountData };
}

2️⃣ Replace manual account creation with enrichment
Update /server/index.js

REMOVE the old /accounts POST route
ADD this one instead

import { enrichAndSave } from "./enrich-and-save.js";

/* ENRICH + CREATE ACCOUNT */
app.post("/accounts/enrich", async (req, res) => {
  try {
    const { domain } = req.body;
    const result = await enrichAndSave(domain);
    res.json(result);
  } catch (e) {
    res.status(400).json({ error: e.message });
  }
});

<!doctype html>
<html>
<head>
  <title>Tony CRM</title>
</head>
<body>
  <h1>Tony CRM – Accounts</h1>
  <ul id="accounts"></ul>

  <script>
    fetch("/accounts")
      .then(r => r.json())
      .then(data => {
        const ul = document.getElementById("accounts");
        data.forEach(a => {
          const li = document.createElement("li");
          li.innerHTML = `
            <a href="account.html?id=${a.id}">
              ${a.company_name} | ${a.tier} | ${a.division_fit}
            </a>
          `;
          ul.appendChild(li);
        });
      });
  </script>
</body>
</html>
